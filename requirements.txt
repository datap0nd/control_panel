$script = "$env:TEMP\install-composer-windows.ps1"
$headers = @{ Authorization = "token $env:PAT_CODE"; "Cache-Control" = "no-cache" }

Invoke-WebRequest `
  -UseBasicParsing `
  -Headers $headers `
  -Uri "https://raw.githubusercontent.com/datap0nd/composer/main/portable/install-composer-windows.ps1?ts=$(Get-Date -Format yyyyMMddHHmmss)" `
  -OutFile $script

powershell -ExecutionPolicy Bypass -File $script
