		measure 'KPI Comparison Data End Date' =
				CALCULATE (
					MAX ( Sales[Reporting Date] ),
					REMOVEFILTERS ()
				)
			formatString: yyyy-mm-dd
			isHidden
			displayFolder: Helper

		measure 'KPI Comparison Period Start Date' =
				VAR _DataStart =
					CALCULATE (
						MIN ( Sales[Reporting Date] ),
						REMOVEFILTERS ()
					)
				VAR _DataEnd = [KPI Comparison Data End Date]
				VAR _HasDateContext = [Policy Date Context Flag] = 1
				VAR _SelectedStart = MIN ( 'Date'[Date] )
				RETURN
					IF (
						ISBLANK ( _DataStart ) || ISBLANK ( _DataEnd ),
						BLANK (),
						IF ( _HasDateContext, MAX ( _SelectedStart, _DataStart ), _DataStart )
					)
			formatString: yyyy-mm-dd
			isHidden
			displayFolder: Helper

		measure 'KPI Comparison Period End Date' =
				VAR _DataEnd = [KPI Comparison Data End Date]
				VAR _HasDateContext = [Policy Date Context Flag] = 1
				VAR _SelectedEnd = MAX ( 'Date'[Date] )
				RETURN
					IF (
						ISBLANK ( _DataEnd ),
						BLANK (),
						IF ( _HasDateContext, MIN ( _SelectedEnd, _DataEnd ), _DataEnd )
					)
			formatString: yyyy-mm-dd
			isHidden
			displayFolder: Helper
