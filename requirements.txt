DEFINE
    VAR __JuneStart = DATE ( 2026, 6, 1 )
    VAR __JuneEnd = DATE ( 2026, 6, 30 )
    VAR __PaymentTypes =
        DISTINCT (
            SELECTCOLUMNS (
                ALL ( Policy ),
                "Payment Type", Policy[Payment Type]
            )
        )

EVALUATE
    ADDCOLUMNS (
        __PaymentTypes,
        "All June Cancellations",
            VAR __PaymentType = [Payment Type]
            RETURN
                COUNTROWS (
                    FILTER (
                        ALL ( Policy ),
                        Policy[Payment Type] = __PaymentType
                            && NOT ISBLANK ( Policy[Cancellation Date] )
                            && Policy[Cancellation Date] >= __JuneStart
                            && Policy[Cancellation Date] <= __JuneEnd
                    )
                ),
        "Positive Amount June Cancellations",
            VAR __PaymentType = [Payment Type]
            RETURN
                COUNTROWS (
                    FILTER (
                        ALL ( Policy ),
                        Policy[Payment Type] = __PaymentType
                            && COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
                            && NOT ISBLANK ( Policy[Cancellation Date] )
                            && Policy[Cancellation Date] >= __JuneStart
                            && Policy[Cancellation Date] <= __JuneEnd
                    )
                ),
        "Distinct Products With Positive Amount",
            VAR __PaymentType = [Payment Type]
            RETURN
                CALCULATE (
                    DISTINCTCOUNT ( Policy[Product ID] ),
                    FILTER (
                        ALL ( Policy ),
                        Policy[Payment Type] = __PaymentType
                            && COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
                            && NOT ISBLANK ( Policy[Cancellation Date] )
                            && Policy[Cancellation Date] >= __JuneStart
                            && Policy[Cancellation Date] <= __JuneEnd
                    )
                ),
        "Minimum Positive Monthly Amount",
            VAR __PaymentType = [Payment Type]
            RETURN
                MINX (
                    FILTER (
                        ALL ( Policy ),
                        Policy[Payment Type] = __PaymentType
                            && COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
                            && NOT ISBLANK ( Policy[Cancellation Date] )
                            && Policy[Cancellation Date] >= __JuneStart
                            && Policy[Cancellation Date] <= __JuneEnd
                    ),
                    Policy[Product Monthly Amount]
                ),
        "Maximum Positive Monthly Amount",
            VAR __PaymentType = [Payment Type]
            RETURN
                MAXX (
                    FILTER (
                        ALL ( Policy ),
                        Policy[Payment Type] = __PaymentType
                            && COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
                            && NOT ISBLANK ( Policy[Cancellation Date] )
                            && Policy[Cancellation Date] >= __JuneStart
                            && Policy[Cancellation Date] <= __JuneEnd
                    ),
                    Policy[Product Monthly Amount]
                )
    )

ORDER BY [Positive Amount June Cancellations] DESC
