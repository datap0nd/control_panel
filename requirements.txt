SELECT
  SALE_CHNL_CD,
  PRTNR_NM,
  SALE_PRTNR_KEY,
  PROD_ID,
  CP_STS_CD,
  PMT_TP_CD,
  SVC_DEL_YN_FG,
  DEL_YN_FG,
  SALE_TP,
  PROD_SALE_TP,
  COUNT(DISTINCT CP_SRL_NUM) AS distinct_policy_count
FROM `bigdata-platform-data.scp_7cu.SMCDB_TCMC_PRTY_SVC_ec1_cntry_AM`
WHERE COALESCE(
    SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS DATE),
    DATE(SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS TIMESTAMP)),
    SAFE.PARSE_DATE(
      '%Y%m%d',
      REGEXP_REPLACE(CAST(INSR_PLCY_CRT_DT AS STRING), r'[^0-9]', '')
    )
  ) = DATE '2026-01-16'
  AND cnty_cd_AM_cntry = 'COUNTRY_CODE'
GROUP BY
  SALE_CHNL_CD,
  PRTNR_NM,
  SALE_PRTNR_KEY,
  PROD_ID,
  CP_STS_CD,
  PMT_TP_CD,
  SVC_DEL_YN_FG,
  DEL_YN_FG,
  SALE_TP,
  PROD_SALE_TP
ORDER BY distinct_policy_count DESC;


SELECT
  CNTY_CD,
  SALE_CHNL_NM,
  SUB_SALE_CHNL_NM,
  PERKS_CTGY,
  SVC_PRDCT_YN_FG,
  DVCE_MODEL_NM,
  PMT_TP_CD,
  PFM_TP,
  SUM(COALESCE(SAFE_CAST(SALE_CNT AS NUMERIC), 0)) AS aggregate_sales_count
FROM `bigdata-platform-data.scp_7cu.SMCDB_TCMC_DSBD_SCP_STAT_CNTRCT_d5f9595b_ec1_cntry_AM`
WHERE COALESCE(
    SAFE_CAST(CAST(STAT_DD AS STRING) AS DATE),
    SAFE.PARSE_DATE(
      '%Y%m%d',
      REGEXP_REPLACE(CAST(STAT_DD AS STRING), r'[^0-9]', '')
    )
  ) = DATE '2026-01-16'
  AND CNTY_CD = 'COUNTRY_CODE'
GROUP BY
  CNTY_CD,
  SALE_CHNL_NM,
  SUB_SALE_CHNL_NM,
  PERKS_CTGY,
  SVC_PRDCT_YN_FG,
  DVCE_MODEL_NM,
  PMT_TP_CD,
  PFM_TP
HAVING aggregate_sales_count <> 0
ORDER BY aggregate_sales_count DESC;
