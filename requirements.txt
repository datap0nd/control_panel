DEFINE
    VAR _DataEnd = [KPI Comparison Data End Date]
    VAR _CY = YEAR ( _DataEnd )
    VAR _LY = _CY - 1
    VAR _LatestMonth = MONTH ( _DataEnd )
    VAR _LatestDay = DAY ( _DataEnd )

EVALUATE
    SUMMARIZECOLUMNS (
        'Date'[MonthNum],
        'Date'[Month],

        // These are the boundaries currently detected inside each chart point.
        "Detected Period Start", [KPI Comparison Period Start Date],
        "Detected Period End", [KPI Comparison Period End Date],

        // Values currently plotted by the three Executive charts.
        "Sales Chart CY", [Sales QTY CY],
        "Sales Chart LY", [Sales QTY LY],
        "Attach Chart CY", [Attach Rate CY],
        "Attach Chart LY", [Attach Rate LY Chart],
        "Margin Chart CY", [Net Revenue CY],
        "Margin Chart LY", [Net Revenue LY Chart],

        // Correct non-cumulative monthly values at the chart grain.
        "Sales Monthly CY",
            VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
            RETURN
                CALCULATE (
                    [Sales Volume],
                    REMOVEFILTERS ( 'Date' ),
                    REMOVEFILTERS ( 'Date Filter' ),
                    'Date'[Year] = _CY,
                    'Date'[MonthNum] = _Month
                ),
        "Sales Monthly LY Full Month",
            VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
            RETURN
                CALCULATE (
                    [Sales Volume],
                    REMOVEFILTERS ( 'Date' ),
                    REMOVEFILTERS ( 'Date Filter' ),
                    'Date'[Year] = _LY,
                    'Date'[MonthNum] = _Month
                ),
        "Sales Monthly LY Comparable Cutoff",
            VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
            VAR _LYMonthStart = DATE ( _LY, _Month, 1 )
            VAR _LYMonthEnd =
                IF (
                    _Month = _LatestMonth,
                    DATE ( _LY, _Month, _LatestDay ),
                    EOMONTH ( _LYMonthStart, 0 )
                )
            RETURN
                IF (
                    _Month > _LatestMonth,
                    BLANK (),
                    CALCULATE (
                        [Sales Volume],
                        REMOVEFILTERS ( 'Date' ),
                        REMOVEFILTERS ( 'Date Filter' ),
                        DATESBETWEEN ( 'Date'[Date], _LYMonthStart, _LYMonthEnd )
                    )
                ),
        "Attach Monthly CY",
            VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
            RETURN
                CALCULATE (
                    [Raw Attach Rate],
                    REMOVEFILTERS ( 'Date' ),
                    REMOVEFILTERS ( 'Date Filter' ),
                    'Date'[Year] = _CY,
                    'Date'[MonthNum] = _Month
                ),
        "Attach Monthly LY Full Month",
            VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
            RETURN
                CALCULATE (
                    [Raw Attach Rate],
                    REMOVEFILTERS ( 'Date' ),
                    REMOVEFILTERS ( 'Date Filter' ),
                    'Date'[Year] = _LY,
                    'Date'[MonthNum] = _Month
                ),
        "Margin Monthly CY",
            VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
            RETURN
                CALCULATE (
                    [Net Revenue / Program Margin],
                    REMOVEFILTERS ( 'Date' ),
                    REMOVEFILTERS ( 'Date Filter' ),
                    'Date'[Year] = _CY,
                    'Date'[MonthNum] = _Month
                ),
        "Margin Monthly LY Full Month",
            VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
            RETURN
                CALCULATE (
                    [Net Revenue / Program Margin],
                    REMOVEFILTERS ( 'Date' ),
                    REMOVEFILTERS ( 'Date Filter' ),
                    'Date'[Year] = _LY,
                    'Date'[MonthNum] = _Month
                ),

        // A non-zero difference proves that the plotted measure is not monthly.
        "Sales CY Difference",
            VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
            VAR _Expected =
                CALCULATE (
                    [Sales Volume],
                    REMOVEFILTERS ( 'Date' ),
                    REMOVEFILTERS ( 'Date Filter' ),
                    'Date'[Year] = _CY,
                    'Date'[MonthNum] = _Month
                )
            RETURN [Sales QTY CY] - _Expected,
        "Margin CY Difference",
            VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
            VAR _Expected =
                CALCULATE (
                    [Net Revenue / Program Margin],
                    REMOVEFILTERS ( 'Date' ),
                    REMOVEFILTERS ( 'Date Filter' ),
                    'Date'[Year] = _CY,
                    'Date'[MonthNum] = _Month
                )
            RETURN [Net Revenue CY] - _Expected
    )

ORDER BY 'Date'[MonthNum]
