/// Paste-ready TMDL createOrReplace script for the full-page Samsung executive-summary HTML Content visual.
/// The single measure returns the title, display-only page tabs, display-only filter context, KPIs, signals, trends, and channel mix.
/// The tabs and filter chips mirror report state only. They do not navigate or set filters. No scripts, links, web fonts, images, or other external assets are used.

createOrReplace
	ref table Measures_table

		measure 'HTML Executive Summary Samsung' =
				VAR _AnalysisYear = [Analysis Year]
				VAR _AsOfDate = COALESCE ( [KPI Executive Sparkline Anchor Date], MAX ( 'Date'[Date] ), TODAY () )
				VAR _OwnerRaw = COALESCE ( [Business Owner], "Not assigned" )
				VAR _OwnerText = SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( _OwnerRaw, "&", "&amp;" ), "<", "&lt;" ), ">", "&gt;" ), "'", "&#39;" ), """", "&quot;" )
				VAR _EvaluatedAt = [Last Refreshed]
				VAR _EvaluatedText = IF ( ISBLANK ( _EvaluatedAt ), "Not evaluated", FORMAT ( _EvaluatedAt, "dd mmm yyyy HH:mm" ) )

				VAR _PeriodLabelFilters = FILTERS ( 'Date Filter'[Period Label] )
				VAR _PeriodLabelCount = COUNTROWS ( _PeriodLabelFilters )
				VAR _PeriodTypeFilters = FILTERS ( 'Date Filter'[Period Type] )
				VAR _PeriodTypeCount = COUNTROWS ( _PeriodTypeFilters )
				VAR _PeriodRaw =
					SWITCH (
						TRUE (),
						_PeriodLabelCount = 1, COALESCE ( MAXX ( _PeriodLabelFilters, 'Date Filter'[Period Label] ), "No period" ),
						_PeriodLabelCount > 1, FORMAT ( _PeriodLabelCount, "#,##0" ) & " periods",
						_PeriodTypeCount = 1, COALESCE ( MAXX ( _PeriodTypeFilters, 'Date Filter'[Period Type] ), "No period type" ),
						_PeriodTypeCount > 1, FORMAT ( _PeriodTypeCount, "#,##0" ) & " period types",
						"All periods"
					)
				VAR _PeriodText = SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( _PeriodRaw, "&", "&amp;" ), "<", "&lt;" ), ">", "&gt;" ), "'", "&#39;" ), """", "&quot;" )

				VAR _SubsidiaryFilters = FILTERS ( 'Subsidiary Master'[Subsidiary] )
				VAR _SubsidiaryCount = COUNTROWS ( _SubsidiaryFilters )
				VAR _SubsidiaryRaw =
					SWITCH (
						TRUE (),
						_SubsidiaryCount = 1, COALESCE ( MAXX ( _SubsidiaryFilters, 'Subsidiary Master'[Subsidiary] ), "No subsidiary" ),
						_SubsidiaryCount > 1, FORMAT ( _SubsidiaryCount, "#,##0" ) & " subsidiaries",
						"All subsidiaries"
					)
				VAR _SubsidiaryText = SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( _SubsidiaryRaw, "&", "&amp;" ), "<", "&lt;" ), ">", "&gt;" ), "'", "&#39;" ), """", "&quot;" )

				VAR _DeviceModelFilters = FILTERS ( Sales[Device Model] )
				VAR _DeviceModelCount = COUNTROWS ( _DeviceModelFilters )
				VAR _DeviceSeriesFilters = FILTERS ( Sales[Device Series] )
				VAR _DeviceSeriesCount = COUNTROWS ( _DeviceSeriesFilters )
				VAR _DeviceFamilyFilters = FILTERS ( Sales[Product Family] )
				VAR _DeviceFamilyCount = COUNTROWS ( _DeviceFamilyFilters )
				VAR _DeviceRaw =
					SWITCH (
						TRUE (),
						_DeviceModelCount = 1, COALESCE ( MAXX ( _DeviceModelFilters, Sales[Device Model] ), "No device" ),
						_DeviceModelCount > 1, FORMAT ( _DeviceModelCount, "#,##0" ) & " device models",
						_DeviceSeriesCount = 1, COALESCE ( MAXX ( _DeviceSeriesFilters, Sales[Device Series] ), "No series" ) & " series",
						_DeviceSeriesCount > 1, FORMAT ( _DeviceSeriesCount, "#,##0" ) & " device series",
						_DeviceFamilyCount = 1, COALESCE ( MAXX ( _DeviceFamilyFilters, Sales[Product Family] ), "No family" ),
						_DeviceFamilyCount > 1, FORMAT ( _DeviceFamilyCount, "#,##0" ) & " device families",
						"All devices"
					)
				VAR _DeviceText = SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( _DeviceRaw, "&", "&amp;" ), "<", "&lt;" ), ">", "&gt;" ), "'", "&#39;" ), """", "&quot;" )

				VAR _PartnerFilters = FILTERS ( Sales[Account / Partner] )
				VAR _PartnerCount = COUNTROWS ( _PartnerFilters )
				VAR _SubChannelFilters = FILTERS ( Sales[Sub Sales Channel] )
				VAR _SubChannelCount = COUNTROWS ( _SubChannelFilters )
				VAR _SalesChannelFilters = FILTERS ( Sales[Sales Channel] )
				VAR _ChannelCountContext = COUNTROWS ( _SalesChannelFilters )
				VAR _ChannelRaw =
					SWITCH (
						TRUE (),
						_PartnerCount = 1, COALESCE ( MAXX ( _PartnerFilters, Sales[Account / Partner] ), "No partner" ),
						_PartnerCount > 1, FORMAT ( _PartnerCount, "#,##0" ) & " partners",
						_SubChannelCount = 1, COALESCE ( MAXX ( _SubChannelFilters, Sales[Sub Sales Channel] ), "No sub-channel" ),
						_SubChannelCount > 1, FORMAT ( _SubChannelCount, "#,##0" ) & " sub-channels",
						_ChannelCountContext = 1, COALESCE ( MAXX ( _SalesChannelFilters, Sales[Sales Channel] ), "No channel" ),
						_ChannelCountContext > 1, FORMAT ( _ChannelCountContext, "#,##0" ) & " channels",
						"All channels"
					)
				VAR _ChannelText = SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( _ChannelRaw, "&", "&amp;" ), "<", "&lt;" ), ">", "&gt;" ), "'", "&#39;" ), """", "&quot;" )

				VAR _ServiceFilters = FILTERS ( Sales[Product Type] )
				VAR _ServiceCount = COUNTROWS ( _ServiceFilters )
				VAR _ServiceRaw =
					SWITCH (
						TRUE (),
						_ServiceCount = 1, COALESCE ( MAXX ( _ServiceFilters, Sales[Product Type] ), "No service type" ),
						_ServiceCount > 1, FORMAT ( _ServiceCount, "#,##0" ) & " service types",
						"All services"
					)
				VAR _ServiceText = SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( _ServiceRaw, "&", "&amp;" ), "<", "&lt;" ), ">", "&gt;" ), "'", "&#39;" ), """", "&quot;" )

				VAR _PaymentFilters = FILTERS ( Sales[Payment Type] )
				VAR _PaymentCount = COUNTROWS ( _PaymentFilters )
				VAR _PaymentRaw =
					SWITCH (
						TRUE (),
						_PaymentCount = 1, COALESCE ( MAXX ( _PaymentFilters, Sales[Payment Type] ), "No payment type" ),
						_PaymentCount > 1, FORMAT ( _PaymentCount, "#,##0" ) & " payment types",
						"All payments"
					)
				VAR _PaymentText = SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( SUBSTITUTE ( _PaymentRaw, "&", "&amp;" ), "<", "&lt;" ), ">", "&gt;" ), "'", "&#39;" ), """", "&quot;" )

				VAR _AttachUnsupportedContext =
					( ( _DeviceFamilyCount > 0 || _DeviceSeriesCount > 0 ) && _DeviceModelCount = 0 )
						|| _PartnerCount > 0
						|| _ServiceCount > 0
						|| _PaymentCount > 0
				VAR _PartnersUnsupportedContext =
					_ChannelCountContext > 0
						|| _SubChannelCount > 0
						|| _PartnerCount > 0
						|| _DeviceFamilyCount > 0
						|| _DeviceSeriesCount > 0
						|| _DeviceModelCount > 0
						|| _ServiceCount > 0
						|| _PaymentCount > 0
				VAR _SparkStart = DATE ( YEAR ( EDATE ( _AsOfDate, -5 ) ), MONTH ( EDATE ( _AsOfDate, -5 ) ), 1 )
				VAR _SparkSeries =
					ADDCOLUMNS (
						GENERATESERIES ( 0, 5, 1 ),
						"__Sales",
							VAR _MonthStart = EDATE ( _SparkStart, [Value] )
							RETURN
								COALESCE (
									CALCULATE (
										[Sales Volume],
										REMOVEFILTERS ( 'Date' ),
										REMOVEFILTERS ( 'Date Filter' ),
										DATESBETWEEN ( 'Date'[Date], _MonthStart, EOMONTH ( _MonthStart, 0 ) )
									),
									0
								),
						"__Attach",
							VAR _MonthStart = EDATE ( _SparkStart, [Value] )
							RETURN
								IF (
									_AttachUnsupportedContext,
									BLANK (),
									COALESCE (
										CALCULATE (
											[Raw Attach Rate],
											REMOVEFILTERS ( 'Date' ),
											REMOVEFILTERS ( 'Date Filter' ),
											DATESBETWEEN ( 'Date'[Date], _MonthStart, EOMONTH ( _MonthStart, 0 ) )
										),
										0
									)
								),
						"__Revenue",
							VAR _MonthStart = EDATE ( _SparkStart, [Value] )
							RETURN
								COALESCE (
									CALCULATE (
										[Gross Revenue],
										REMOVEFILTERS ( 'Date' ),
										REMOVEFILTERS ( 'Date Filter' ),
										DATESBETWEEN ( 'Date'[Date], _MonthStart, EOMONTH ( _MonthStart, 0 ) )
									),
									0
								),
						"__Policies",
							VAR _MonthStart = EDATE ( _SparkStart, [Value] )
							RETURN
								COALESCE (
									CALCULATE (
										[Active Policies],
										REMOVEFILTERS ( 'Date' ),
										REMOVEFILTERS ( 'Date Filter' ),
										DATESBETWEEN ( 'Date'[Date], _MonthStart, EOMONTH ( _MonthStart, 0 ) )
									),
									0
								)
					)
				VAR _SalesSparkMin = MINX ( _SparkSeries, [__Sales] )
				VAR _SalesSparkMax = MAXX ( _SparkSeries, [__Sales] )
				VAR _SalesSparkRange = IF ( _SalesSparkMax = _SalesSparkMin, 1, _SalesSparkMax - _SalesSparkMin )
				VAR _SalesSpark =
					CONCATENATEX (
						_SparkSeries,
						VAR _X = 4 + [Value] * 22.4
						VAR _Y = 5 + DIVIDE ( _SalesSparkMax - [__Sales], _SalesSparkRange, 0.5 ) * 30
						RETURN FORMAT ( _X, "0.0" ) & "," & FORMAT ( _Y, "0.0" ),
						" ",
						[Value],
						ASC
					)
				VAR _AttachSparkMin = MINX ( _SparkSeries, [__Attach] )
				VAR _AttachSparkMax = MAXX ( _SparkSeries, [__Attach] )
				VAR _AttachSparkRange = IF ( _AttachSparkMax = _AttachSparkMin, 1, _AttachSparkMax - _AttachSparkMin )
				VAR _AttachSpark =
					IF (
						_AttachUnsupportedContext,
						"",
						CONCATENATEX (
							_SparkSeries,
							VAR _X = 4 + [Value] * 22.4
							VAR _Y = 5 + DIVIDE ( _AttachSparkMax - [__Attach], _AttachSparkRange, 0.5 ) * 30
							RETURN FORMAT ( _X, "0.0" ) & "," & FORMAT ( _Y, "0.0" ),
							" ",
							[Value],
							ASC
						)
					)
				VAR _RevenueSparkMin = MINX ( _SparkSeries, [__Revenue] )
				VAR _RevenueSparkMax = MAXX ( _SparkSeries, [__Revenue] )
				VAR _RevenueSparkRange = IF ( _RevenueSparkMax = _RevenueSparkMin, 1, _RevenueSparkMax - _RevenueSparkMin )
				VAR _RevenueSpark =
					CONCATENATEX (
						_SparkSeries,
						VAR _X = 4 + [Value] * 22.4
						VAR _Y = 5 + DIVIDE ( _RevenueSparkMax - [__Revenue], _RevenueSparkRange, 0.5 ) * 30
						RETURN FORMAT ( _X, "0.0" ) & "," & FORMAT ( _Y, "0.0" ),
						" ",
						[Value],
						ASC
					)
				VAR _PolicySparkMin = MINX ( _SparkSeries, [__Policies] )
				VAR _PolicySparkMax = MAXX ( _SparkSeries, [__Policies] )
				VAR _PolicySparkRange = IF ( _PolicySparkMax = _PolicySparkMin, 1, _PolicySparkMax - _PolicySparkMin )
				VAR _PolicySpark =
					CONCATENATEX (
						_SparkSeries,
						VAR _X = 4 + [Value] * 22.4
						VAR _Y = 5 + DIVIDE ( _PolicySparkMax - [__Policies], _PolicySparkRange, 0.5 ) * 30
						RETURN FORMAT ( _X, "0.0" ) & "," & FORMAT ( _Y, "0.0" ),
						" ",
						[Value],
						ASC
					)

				VAR _Sales = [Sales Volume]
				VAR _SalesLy = [Sales Volume LY]
				VAR _SalesPlan = [Sales Plan QTY]
				VAR _SalesLyDelta = DIVIDE ( _Sales - _SalesLy, _SalesLy )
				VAR _SalesPlanRatio = DIVIDE ( _Sales, _SalesPlan )
				VAR _SalesText = IF ( ISBLANK ( _Sales ), "--", FORMAT ( _Sales, "#,##0" ) )
				VAR _SalesLyText = IF ( ISBLANK ( _SalesLyDelta ), "n/a", FORMAT ( _SalesLyDelta, "+0.0%;-0.0%;0.0%" ) )
				VAR _SalesPlanText = IF ( ISBLANK ( _SalesPlanRatio ), "n/a", FORMAT ( _SalesPlanRatio, "0.0%" ) )
				VAR _SalesLyClass = IF ( ISBLANK ( _SalesLyDelta ), "na", IF ( _SalesLyDelta >= 0, "good", "bad" ) )
				VAR _SalesPlanClass = IF ( ISBLANK ( _SalesPlanRatio ), "na", IF ( _SalesPlanRatio >= 1, "good", "warning" ) )

				VAR _Attach = IF ( _AttachUnsupportedContext, BLANK (), [Raw Attach Rate] )
				VAR _AttachLy = IF ( _AttachUnsupportedContext, BLANK (), [Raw Attach Rate LY] )
				VAR _AttachPlan = IF ( _AttachUnsupportedContext, BLANK (), [Sales Plan Attach Rate] )
				VAR _AttachLyPoints = ( _Attach - _AttachLy ) * 100
				VAR _AttachPlanPoints = ( _Attach - _AttachPlan ) * 100
				VAR _AttachText = IF ( ISBLANK ( _Attach ), "--", FORMAT ( _Attach, "0.0%" ) )
				VAR _AttachLyText = IF ( ISBLANK ( _Attach ) || ISBLANK ( _AttachLy ), "n/a", FORMAT ( _AttachLyPoints, "+0.0 pts;-0.0 pts;0.0 pts" ) )
				VAR _AttachPlanText = IF ( ISBLANK ( _Attach ) || ISBLANK ( _AttachPlan ), "n/a", FORMAT ( _AttachPlanPoints, "+0.0 pts;-0.0 pts;0.0 pts" ) )
				VAR _AttachLyClass = IF ( ISBLANK ( _Attach ) || ISBLANK ( _AttachLy ), "na", IF ( _AttachLyPoints >= 0, "good", "bad" ) )
				VAR _AttachPlanClass = IF ( ISBLANK ( _Attach ) || ISBLANK ( _AttachPlan ), "na", IF ( _AttachPlanPoints >= 0, "good", "warning" ) )

				VAR _Revenue = [Gross Revenue]
				VAR _RevenueLy = [Gross Revenue LY]
				VAR _RevenuePlan = [Revenue Plan Amount]
				VAR _RevenueLyDelta = DIVIDE ( _Revenue - _RevenueLy, _RevenueLy )
				VAR _RevenuePlanRatio = DIVIDE ( _Revenue, _RevenuePlan )
				VAR _RevenueText =
					SWITCH (
						TRUE (),
						ISBLANK ( _Revenue ), "--",
						ABS ( _Revenue ) >= 1000000000, FORMAT ( DIVIDE ( _Revenue, 1000000000 ), "0.0" ) & "B",
						ABS ( _Revenue ) >= 1000000, FORMAT ( DIVIDE ( _Revenue, 1000000 ), "0.0" ) & "M",
						ABS ( _Revenue ) >= 1000, FORMAT ( DIVIDE ( _Revenue, 1000 ), "0.0" ) & "K",
						FORMAT ( _Revenue, "#,##0" )
					)
				VAR _RevenueLyText = IF ( ISBLANK ( _RevenueLyDelta ), "n/a", FORMAT ( _RevenueLyDelta, "+0.0%;-0.0%;0.0%" ) )
				VAR _RevenuePlanText = IF ( ISBLANK ( _RevenuePlanRatio ), "n/a", FORMAT ( _RevenuePlanRatio, "0.0%" ) )
				VAR _RevenueLyClass = IF ( ISBLANK ( _RevenueLyDelta ), "na", IF ( _RevenueLyDelta >= 0, "good", "bad" ) )
				VAR _RevenuePlanClass = IF ( ISBLANK ( _RevenuePlanRatio ), "na", IF ( _RevenuePlanRatio >= 1, "good", "warning" ) )

				VAR _ActivePolicies = [Active Policies]
				VAR _ActivePoliciesLy =
					CALCULATE (
						[Active Policies],
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						TREATAS (
							SELECTCOLUMNS ( VALUES ( 'Date'[Date] ), "Date", EDATE ( 'Date'[Date], -12 ) ),
							'Date'[Date]
						)
					)
				VAR _ActivePoliciesDelta = DIVIDE ( _ActivePolicies - _ActivePoliciesLy, _ActivePoliciesLy )
				VAR _ActivePoliciesText = IF ( ISBLANK ( _ActivePolicies ), "--", FORMAT ( _ActivePolicies, "#,##0" ) )
				VAR _ActivePoliciesLyText = IF ( ISBLANK ( _ActivePoliciesDelta ), "n/a", FORMAT ( _ActivePoliciesDelta, "+0.0%;-0.0%;0.0%" ) )
				VAR _ActivePoliciesLyClass = IF ( ISBLANK ( _ActivePoliciesDelta ), "na", IF ( _ActivePoliciesDelta >= 0, "good", "bad" ) )

				VAR _Partners = IF ( _PartnersUnsupportedContext, BLANK (), [Active Channel Partners] )
				VAR _PartnersLy =
					IF (
						_PartnersUnsupportedContext,
						BLANK (),
						CALCULATE (
							[Active Channel Partners],
							REMOVEFILTERS ( 'Date' ),
							REMOVEFILTERS ( 'Date Filter' ),
							TREATAS (
								SELECTCOLUMNS ( VALUES ( 'Date'[Date] ), "Date", EDATE ( 'Date'[Date], -12 ) ),
								'Date'[Date]
							)
						)
					)
				VAR _PartnersChange = _Partners - _PartnersLy
				VAR _PartnersText = IF ( _PartnersUnsupportedContext, "n/a", IF ( ISBLANK ( _Partners ), "--", FORMAT ( _Partners, "#,##0" ) ) )
				VAR _PartnersChangeText =
					SWITCH (
						TRUE (),
						_PartnersUnsupportedContext, "Unavailable",
						ISBLANK ( _PartnersLy ), "n/a",
						_PartnersChange = 0, "No change",
						FORMAT ( _PartnersChange, "+#,##0;-#,##0;0" ) & " vs LY"
					)
				VAR _PartnersClass = IF ( _PartnersUnsupportedContext || ISBLANK ( _PartnersLy ) || _PartnersChange = 0, "neutral", IF ( _PartnersChange > 0, "good", "warning" ) )

				VAR _Subscribers = [Active Monthly Subscribers]
				VAR _SubscribersLy = [Monthly Paid Revenue Units LY]
				VAR _SubscribersDelta = DIVIDE ( _Subscribers - _SubscribersLy, _SubscribersLy )
				VAR _SubscribersText = IF ( ISBLANK ( _Subscribers ), "--", FORMAT ( _Subscribers, "#,##0" ) )
				VAR _SubscribersDeltaText = IF ( ISBLANK ( _SubscribersDelta ), "n/a", FORMAT ( _SubscribersDelta, "+0.0%;-0.0%;0.0%" ) )
				VAR _SubscribersClass = IF ( ISBLANK ( _SubscribersDelta ), "neutral", IF ( _SubscribersDelta >= 0, "good", "warning" ) )

				VAR _MonthlyCancel = [Monthly Cancellation Rate]
				VAR _MonthlyCancelLy =
					CALCULATE (
						[Monthly Cancellation Rate],
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						TREATAS (
							SELECTCOLUMNS ( VALUES ( 'Date'[Date] ), "Date", EDATE ( 'Date'[Date], -12 ) ),
							'Date'[Date]
						)
					)
				VAR _MonthlyCancelPoints = ( _MonthlyCancel - _MonthlyCancelLy ) * 100
				VAR _MonthlyCancelText = IF ( ISBLANK ( _MonthlyCancel ), "--", FORMAT ( _MonthlyCancel, "0.0%" ) )
				VAR _MonthlyCancelDeltaText = IF ( ISBLANK ( _MonthlyCancel ) || ISBLANK ( _MonthlyCancelLy ), "n/a", FORMAT ( _MonthlyCancelPoints, "+0.0 pts;-0.0 pts;0.0 pts" ) )
				VAR _MonthlyCancelClass = IF ( ISBLANK ( _MonthlyCancel ) || ISBLANK ( _MonthlyCancelLy ), "neutral", IF ( _MonthlyCancelPoints <= 0, "good", "warning" ) )

				VAR _MonthAxis =
					FILTER (
						SUMMARIZE ( ALLSELECTED ( 'Date' ), 'Date'[MonthNum], 'Date'[Month] ),
						NOT ISBLANK ( 'Date'[MonthNum] )
					)
				VAR _SalesSeries =
					ADDCOLUMNS (
						_MonthAxis,
						"__Label", 'Date'[Month],
						"__CY", CALCULATE ( [Sales QTY CY] ),
						"__LY", CALCULATE ( [Sales QTY LY] )
					)
				VAR _SalesVisible = FILTER ( _SalesSeries, NOT ISBLANK ( [__CY] ) || NOT ISBLANK ( [__LY] ) )
				VAR _SalesValues =
					FILTER (
						UNION (
							SELECTCOLUMNS ( _SalesVisible, "__V", [__CY] ),
							SELECTCOLUMNS ( _SalesVisible, "__V", [__LY] )
						),
						NOT ISBLANK ( [__V] )
					)
				VAR _SalesMin = MINX ( _SalesValues, [__V] )
				VAR _SalesMax = MAXX ( _SalesValues, [__V] )
				VAR _SalesRange = IF ( _SalesMax = _SalesMin, IF ( _SalesMax = 0, 1, ABS ( _SalesMax ) ), _SalesMax - _SalesMin )
				VAR _SalesCount = COUNTROWS ( _SalesVisible )
				VAR _SalesPlot =
					ADDCOLUMNS (
						_SalesVisible,
						"__Index", RANKX ( _SalesVisible, 'Date'[MonthNum],, ASC, DENSE ),
						"__X", 48 + DIVIDE ( RANKX ( _SalesVisible, 'Date'[MonthNum],, ASC, DENSE ) - 1, MAX ( 1, _SalesCount - 1 ) ) * 492,
						"__Y1", IF ( ISBLANK ( [__CY] ), BLANK (), 18 + DIVIDE ( _SalesMax - [__CY], _SalesRange, 0.5 ) * 128 ),
						"__Y2", IF ( ISBLANK ( [__LY] ), BLANK (), 18 + DIVIDE ( _SalesMax - [__LY], _SalesRange, 0.5 ) * 128 )
					)
				VAR _SalesLabelStep = MAX ( 1, CEILING ( DIVIDE ( _SalesCount, 6 ), 1 ) )
				VAR _SalesLabels =
					CONCATENATEX (
						FILTER ( _SalesPlot, [__Index] = 1 || [__Index] = _SalesCount || MOD ( [__Index] - 1, _SalesLabelStep ) = 0 ),
						"<text class='axis-label' x='" & FORMAT ( ROUND ( [__X], 0 ), "0" ) & "' y='169' text-anchor='middle'>" & LEFT ( [__Label], 3 ) & "</text>",
						"",
						'Date'[MonthNum],
						ASC
					)
				VAR _SalesCyPath = CONCATENATEX ( FILTER ( _SalesPlot, NOT ISBLANK ( [__CY] ) ), FORMAT ( ROUND ( [__X], 0 ), "0" ) & "," & FORMAT ( ROUND ( [__Y1], 0 ), "0" ), " ", 'Date'[MonthNum], ASC )
				VAR _SalesLyPath = CONCATENATEX ( FILTER ( _SalesPlot, NOT ISBLANK ( [__LY] ) ), FORMAT ( ROUND ( [__X], 0 ), "0" ) & "," & FORMAT ( ROUND ( [__Y2], 0 ), "0" ), " ", 'Date'[MonthNum], ASC )
				VAR _SalesDots =
					CONCATENATEX (
						FILTER ( _SalesPlot, NOT ISBLANK ( [__CY] ) ),
						"<circle class='hit' cx='" & FORMAT ( ROUND ( [__X], 0 ), "0" ) & "' cy='" & FORMAT ( ROUND ( [__Y1], 0 ), "0" ) & "' r='7' fill='transparent'><title>" & [__Label] & " " & FORMAT ( _AnalysisYear, "0" ) & ": " & FORMAT ( [__CY], "#,##0" ) & "</title></circle>" &
						"<circle class='series-dot' cx='" & FORMAT ( ROUND ( [__X], 0 ), "0" ) & "' cy='" & FORMAT ( ROUND ( [__Y1], 0 ), "0" ) & "' r='2.7'/>",
						"",
						'Date'[MonthNum],
						ASC
					)

				VAR _AttachSeries =
					ADDCOLUMNS (
						_MonthAxis,
						"__Label", 'Date'[Month],
						"__CY", IF ( _AttachUnsupportedContext, BLANK (), CALCULATE ( [Attach Rate CY] ) ),
						"__LY", IF ( _AttachUnsupportedContext, BLANK (), CALCULATE ( [Attach Rate LY Chart] ) )
					)
				VAR _AttachVisible = FILTER ( _AttachSeries, NOT ISBLANK ( [__CY] ) || NOT ISBLANK ( [__LY] ) )
				VAR _AttachValues =
					FILTER (
						UNION (
							SELECTCOLUMNS ( _AttachVisible, "__V", [__CY] ),
							SELECTCOLUMNS ( _AttachVisible, "__V", [__LY] )
						),
						NOT ISBLANK ( [__V] )
					)
				VAR _AttachMin = MINX ( _AttachValues, [__V] )
				VAR _AttachMax = MAXX ( _AttachValues, [__V] )
				VAR _AttachRange = IF ( _AttachMax = _AttachMin, IF ( _AttachMax = 0, 1, ABS ( _AttachMax ) ), _AttachMax - _AttachMin )
				VAR _AttachCount = COUNTROWS ( _AttachVisible )
				VAR _AttachPlot =
					ADDCOLUMNS (
						_AttachVisible,
						"__Index", RANKX ( _AttachVisible, 'Date'[MonthNum],, ASC, DENSE ),
						"__X", 48 + DIVIDE ( RANKX ( _AttachVisible, 'Date'[MonthNum],, ASC, DENSE ) - 1, MAX ( 1, _AttachCount - 1 ) ) * 492,
						"__Y1", IF ( ISBLANK ( [__CY] ), BLANK (), 18 + DIVIDE ( _AttachMax - [__CY], _AttachRange, 0.5 ) * 128 ),
						"__Y2", IF ( ISBLANK ( [__LY] ), BLANK (), 18 + DIVIDE ( _AttachMax - [__LY], _AttachRange, 0.5 ) * 128 )
					)
				VAR _AttachLabelStep = MAX ( 1, CEILING ( DIVIDE ( _AttachCount, 6 ), 1 ) )
