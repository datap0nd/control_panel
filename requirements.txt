DEFINE
    VAR _SalesBase =
        FILTER (
            SELECTCOLUMNS (
                ALL ( Sales ),
                "Month Key", COALESCE ( Sales[Month Key], "__BLANK__" ),
                "Country Code", COALESCE ( Sales[Country Code], "__BLANK__" ),
                "Fact Sub", COALESCE ( Sales[Subsidiary], "__BLANK__" ),
                "Master Sub", COALESCE ( RELATED ( 'Subsidiary Master'[Subsidiary] ), "__UNMAPPED__" ),
                "Join Sub",
                    COALESCE (
                        RELATED ( 'Subsidiary Master'[Subsidiary] ),
                        Sales[Subsidiary],
                        "__UNMAPPED__"
                    ),
                "Raw Channel", COALESCE ( Sales[Sales Channel], "__BLANK__" ),
                "Normalized Channel",
                    VAR _Raw = UPPER ( TRIM ( COALESCE ( Sales[Sales Channel], "" ) ) )
                    RETURN SUBSTITUTE ( SUBSTITUTE ( _Raw, "-", "" ), " ", "" ),
                "Channel Bucket",
                    VAR _Raw = UPPER ( TRIM ( COALESCE ( Sales[Sales Channel], "" ) ) )
                    VAR _Normalized = SUBSTITUTE ( SUBSTITUTE ( _Raw, "-", "" ), " ", "" )
                    RETURN IF ( CONTAINSSTRING ( _Normalized, "ESTORE" ), "eStore", "Others" ),
                "Account", COALESCE ( Sales[Account / Partner], "__BLANK__" ),
                "Device Category", COALESCE ( Sales[Device Category], "__BLANK__" ),
                "Device Model", COALESCE ( Sales[Device Model], "__BLANK__" ),
                "Sales Units", COALESCE ( Sales[sales_volume], 0 )
            ),
            [Month Key] <> "__BLANK__"
        )

    VAR _SelloutBase =
        FILTER (
            SELECTCOLUMNS (
                ALL ( 'Sellout Stays' ),
                "Month Key", COALESCE ( 'Sellout Stays'[Month Key], "__BLANK__" ),
                "Country Code", COALESCE ( 'Sellout Stays'[Country Code], "__BLANK__" ),
                "Fact Sub", COALESCE ( 'Sellout Stays'[Subsidiary], "__BLANK__" ),
                "Master Sub", COALESCE ( RELATED ( 'Subsidiary Master'[Subsidiary] ), "__UNMAPPED__" ),
                "Join Sub",
                    COALESCE (
                        RELATED ( 'Subsidiary Master'[Subsidiary] ),
                        'Sellout Stays'[Subsidiary],
                        "__UNMAPPED__"
                    ),
                "Raw Channel", COALESCE ( 'Sellout Stays'[Sales Channel], "__BLANK__" ),
                "Normalized Channel",
                    VAR _Raw = UPPER ( TRIM ( COALESCE ( 'Sellout Stays'[Sales Channel], "" ) ) )
                    RETURN SUBSTITUTE ( SUBSTITUTE ( _Raw, "-", "" ), " ", "" ),
                "Channel Bucket",
                    VAR _Raw = UPPER ( TRIM ( COALESCE ( 'Sellout Stays'[Sales Channel], "" ) ) )
                    VAR _Normalized = SUBSTITUTE ( SUBSTITUTE ( _Raw, "-", "" ), " ", "" )
                    RETURN IF ( CONTAINSSTRING ( _Normalized, "ESTORE" ), "eStore", "Others" ),
                "Account", COALESCE ( 'Sellout Stays'[Account / Partner], "__BLANK__" ),
                "Device Category", COALESCE ( 'Sellout Stays'[Device Category], "__BLANK__" ),
                "Device Model", COALESCE ( 'Sellout Stays'[Device Model], "__BLANK__" ),
                "Sellout Stays Units", COALESCE ( 'Sellout Stays'[so_that_stays_units], 0 ),
                "Sellout Total Units", COALESCE ( 'Sellout Stays'[active_units_resolved], 0 )
            ),
            [Month Key] <> "__BLANK__"
        )

    VAR _SalesChannelProfile =
        GROUPBY (
            _SalesBase,
            [Raw Channel],
            [Normalized Channel],
            [Channel Bucket],
            "Rows", COUNTX ( CURRENTGROUP (), 1 ),
            "Units", SUMX ( CURRENTGROUP (), [Sales Units] )
        )

    VAR _SelloutChannelProfile =
        GROUPBY (
            _SelloutBase,
            [Raw Channel],
            [Normalized Channel],
            [Channel Bucket],
            "Rows", COUNTX ( CURRENTGROUP (), 1 ),
            "Units", SUMX ( CURRENTGROUP (), [Sellout Stays Units] )
        )

    VAR _SalesSubProfile =
        GROUPBY (
            _SalesBase,
            [Country Code],
            [Fact Sub],
            [Master Sub],
            [Join Sub],
            "Rows", COUNTX ( CURRENTGROUP (), 1 ),
            "Units", SUMX ( CURRENTGROUP (), [Sales Units] )
        )

    VAR _SelloutSubProfile =
        GROUPBY (
            _SelloutBase,
            [Country Code],
            [Fact Sub],
            [Master Sub],
            [Join Sub],
            "Rows", COUNTX ( CURRENTGROUP (), 1 ),
            "Units", SUMX ( CURRENTGROUP (), [Sellout Stays Units] )
        )

    VAR _SalesTargetGrain =
        GROUPBY (
            _SalesBase,
            [Month Key],
            [Join Sub],
            [Channel Bucket],
            "Sales Rows", COUNTX ( CURRENTGROUP (), 1 ),
            "Sales Units", SUMX ( CURRENTGROUP (), [Sales Units] )
        )

    VAR _SelloutTargetGrain =
        GROUPBY (
            _SelloutBase,
            [Month Key],
            [Join Sub],
            [Channel Bucket],
            "Sellout Rows", COUNTX ( CURRENTGROUP (), 1 ),
            "Sellout Stays Units", SUMX ( CURRENTGROUP (), [Sellout Stays Units] ),
            "Sellout Total Units", SUMX ( CURRENTGROUP (), [Sellout Total Units] )
        )

    VAR _AllTargetKeys =
        DISTINCT (
            UNION (
                SELECTCOLUMNS (
                    _SalesTargetGrain,
                    "Month Key", [Month Key],
                    "Join Sub", [Join Sub],
                    "Channel Bucket", [Channel Bucket]
                ),
                SELECTCOLUMNS (
                    _SelloutTargetGrain,
                    "Month Key", [Month Key],
                    "Join Sub", [Join Sub],
                    "Channel Bucket", [Channel Bucket]
                )
            )
        )

    VAR _TargetAudit =
        ADDCOLUMNS (
            _AllTargetKeys,
            "Sales Rows",
                VAR _Month = [Month Key]
                VAR _Sub = [Join Sub]
                VAR _Bucket = [Channel Bucket]
                RETURN
                    MAXX (
                        FILTER (
                            _SalesTargetGrain,
                            [Month Key] = _Month
                                && [Join Sub] = _Sub
                                && [Channel Bucket] = _Bucket
                        ),
                        [Sales Rows]
                    ),
            "Sales Units",
                VAR _Month = [Month Key]
                VAR _Sub = [Join Sub]
                VAR _Bucket = [Channel Bucket]
                RETURN
                    MAXX (
                        FILTER (
                            _SalesTargetGrain,
                            [Month Key] = _Month
                                && [Join Sub] = _Sub
                                && [Channel Bucket] = _Bucket
                        ),
                        [Sales Units]
                    ),
            "Sellout Rows",
                VAR _Month = [Month Key]
                VAR _Sub = [Join Sub]
                VAR _Bucket = [Channel Bucket]
                RETURN
                    MAXX (
                        FILTER (
                            _SelloutTargetGrain,
                            [Month Key] = _Month
                                && [Join Sub] = _Sub
                                && [Channel Bucket] = _Bucket
                        ),
                        [Sellout Rows]
                    ),
            "Sellout Stays Units",
                VAR _Month = [Month Key]
                VAR _Sub = [Join Sub]
                VAR _Bucket = [Channel Bucket]
                RETURN
                    MAXX (
                        FILTER (
                            _SelloutTargetGrain,
                            [Month Key] = _Month
                                && [Join Sub] = _Sub
                                && [Channel Bucket] = _Bucket
                        ),
                        [Sellout Stays Units]
                    ),
            "Sellout Total Units",
                VAR _Month = [Month Key]
                VAR _Sub = [Join Sub]
                VAR _Bucket = [Channel Bucket]
                RETURN
                    MAXX (
                        FILTER (
                            _SelloutTargetGrain,
                            [Month Key] = _Month
                                && [Join Sub] = _Sub
                                && [Channel Bucket] = _Bucket
                        ),
                        [Sellout Total Units]
                    )
        )

    VAR _TargetOutput =
        SELECTCOLUMNS (
            _TargetAudit,
            "Audit Section", "TARGET GRAIN COVERAGE",
            "Source", "Sales + Sellout",
            "Month Key", [Month Key],
            "Join Sub", [Join Sub],
            "Master Sub", BLANK (),
            "Fact Sub", BLANK (),
            "Country Code", BLANK (),
            "Raw Channel", BLANK (),
            "Normalized Channel", BLANK (),
            "Channel Bucket", [Channel Bucket],
            "Grain", "Month + Sub + Channel Bucket",
            "Rows", BLANK (),
            "Units", BLANK (),
            "Sales Rows", [Sales Rows],
            "Sales Units", [Sales Units],
            "Sellout Rows", [Sellout Rows],
            "Sellout Stays Units", [Sellout Stays Units],
            "Sellout Total Units", [Sellout Total Units],
            "Attach Rate - Stays", DIVIDE ( [Sales Units], [Sellout Stays Units] ),
            "Attach Rate - Total", DIVIDE ( [Sales Units], [Sellout Total Units] ),
            "Key Status",
                SWITCH (
                    TRUE (),
                    ISBLANK ( [Sales Rows] ), "SELLOUT_ONLY",
                    ISBLANK ( [Sellout Rows] ), "SALES_ONLY",
                    COALESCE ( [Sellout Stays Units], 0 ) = 0, "MATCHED_ZERO_STAYS_DENOMINATOR",
                    "MATCHED"
                )
        )

                "Grain", BLANK (),
                "Rows", [Rows],
                "Units", [Units],
                "Sales Rows", BLANK (),
                "Sales Units", BLANK (),
                "Sellout Rows", BLANK (),
