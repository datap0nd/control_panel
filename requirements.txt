					[Monthly Cancellation Events - Selected Months],
					[Monthly Opening Subscription Exposure - Selected Months]
				)
			formatString: 0.0%
			displayFolder: Sales\Policy Status

		measure 'HTML KPI Monthly Cancellation Rate' =
				VAR _AsOfDate = COALESCE ( [KPI Sales Sparkline Anchor Date], MAX ( 'Date'[Date] ), TODAY () )
				VAR _SparkStart = DATE ( YEAR ( EDATE ( _AsOfDate, -4 ) ), MONTH ( EDATE ( _AsOfDate, -4 ) ), 1 )
				VAR _SparkEnd = DATE ( YEAR ( _AsOfDate ), MONTH ( _AsOfDate ), 1 )
				VAR _StartLabel = FORMAT ( _SparkStart, "MM/yy" )
				VAR _EndLabel = FORMAT ( _SparkEnd, "MM/yy" )

				VAR _SparkTemplatePrefix = "<div class=""spark-wrap""><svg class=""spark"" viewBox=""0 0 156 42"" preserveAspectRatio=""none"" role=""img""><polygon points=""0,42 "
				VAR _SparkTemplateMiddle = " 156,42"" fill=""#e8f1f9""></polygon><polyline points="""
				VAR _SparkTemplateSuffix = """ fill=""none"" stroke=""#1f4e79"" stroke-width=""2.2"" stroke-linecap=""round"" stroke-linejoin=""round""></polyline></svg><div class=""axis""><span>" & _StartLabel & "</span><span>" & _EndLabel & "</span></div></div>"
				VAR _Series =
					ADDCOLUMNS (
						GENERATESERIES ( 0, 4, 1 ),
						"MetricValue",
							VAR _MonthStart = EDATE ( _SparkStart, [Value] )
							RETURN COALESCE ( CALCULATE ( [Monthly Cancellation Rate], REMOVEFILTERS ( 'Date' ), REMOVEFILTERS ( 'Date Filter' ), DATESBETWEEN ( 'Date'[Date], _MonthStart, EOMONTH ( _MonthStart, 0 ) ) ), 0 )

				VAR _MinValue = MINX ( _Series, [MetricValue] )
				VAR _MaxValue = MAXX ( _Series, [MetricValue] )

				VAR _Points =
					CONCATENATEX (

						VAR _X = [Value] * 39
						VAR _Y = 5 + DIVIDE ( _MaxValue - [MetricValue], _Range, 0 ) * 30

						[Monthly Cancellation Rate],

					)
				VAR _Comp1Value = _RawValue - _Prior1Value
				VAR _Comp1Class = IF ( ISBLANK ( _Comp1Value ), "na", IF ( _Comp1Value <= 0, "good", "bad" ) )
				VAR _Comp1Text = IF ( ISBLANK ( _Comp1Value ), "n/a", FORMAT ( _Comp1Value * 100, "+0.0 pts;-0.0 pts;0.0 pts" ) )
				VAR _Comp2Value = _RawValue - _Prior2Value
				VAR _Comp2Class = IF ( ISBLANK ( _Comp2Value ), "na", IF ( _Comp2Value <= 0, "good", "bad" ) )
				VAR _Comp2Text = IF ( ISBLANK ( _Comp2Value ), "n/a", FORMAT ( _Comp2Value * 100, "+0.0 pts;-0.0 pts;0.0 pts" ) )

				VAR _Comp3Text = "n/a"
				VAR _Comp3Label = "No Plan"
