left join (
    select
        s."Series"::text as series,
        s."Category"::text as category,
        case
            when nullif(replace(s."Total"::text, ',', ''), '')::numeric > 1600 then g.global_asp
            when nullif(replace(s."Total"::text, ',', ''), '')::numeric <= 0 then g.global_asp
            else nullif(replace(s."Total"::text, ',', ''), '')::numeric
        end as series_asp
    from "Zahab"."SIBP_CI_Series_ASP" s
    left join (
        select
            "Biz Group"::text as biz_grp,
            "Category"::text as category,
            avg(nullif(replace("Total"::text, ',', ''), '')::numeric) as global_asp
        from "Zahab"."Sell-out Biz Plan Global_ASP"
        where "Category" <> 'DELTA'
        group by "Biz Group", "Category"
    ) g
        on g.biz_grp = s."Biz Group"
       and g.category = s."Category"::text
    where s."Category"::text <> 'DELTA'
      and s."Biz Group" <> 'Total'
      and s."Series" not in ('Series', 'Total')
) sa
    on sa.series = q."Series"
   and sa.category = q."Year"
