										RETURN
											Policy[Paid Monthly Subscriber Flag] = 1
												&& NOT ISBLANK ( _CoverageStart )
												&& _CoverageStart <= _MonthStart
												&& ( ISBLANK ( _WarrantyEnd ) || _WarrantyEnd >= _MonthStart )
												&& ( ISBLANK ( _CancellationDate ) || _CancellationDate > _MonthStart )
									)
								)
						)
					)
			formatString: #,##0
			isHidden
			displayFolder: Helper

		measure 'Monthly Cancellation Rate' =
				DIVIDE ( [Paid Monthly Cancellation Events], [Paid Monthly Opening Exposure] )
			formatString: 0.0%
			displayFolder: SC+ Plan Classification

		measure 'Monthly Cancellation Rate LY' =
				VAR _PeriodStart = [Policy Period Start Date]
				VAR _PeriodEnd = [Policy Period End Date]
				RETURN
					IF (
						ISBLANK ( _PeriodStart ) || ISBLANK ( _PeriodEnd ),
						BLANK (),
						CALCULATE (
							[Monthly Cancellation Rate],
							REMOVEFILTERS ( 'Date' ),
							REMOVEFILTERS ( 'Date Filter' ),
							DATESBETWEEN ( 'Date'[Date], EDATE ( _PeriodStart, -12 ), EDATE ( _PeriodEnd, -12 ) )
						)
					)
			formatString: 0.0%
			displayFolder: SC+ Plan Classification

		measure 'Active Channel Partners' =
				VAR _PeriodStart = [Policy Period Start Date]
				VAR _PeriodEnd = [Policy Period End Date]
				VAR _PolicyRows =
					CALCULATETABLE (
						Policy,
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						REMOVEFILTERS ( Policy[Policy Created Date] )
					)
				VAR _PartnerValues =
					FILTER (
						CALCULATETABLE (
							VALUES ( Policy[Partner] ),
							REMOVEFILTERS ( 'Date' ),
							REMOVEFILTERS ( 'Date Filter' ),
							REMOVEFILTERS ( Policy[Policy Created Date] )
						),
						NOT ISBLANK ( Policy[Partner] )
							&& TRIM ( Policy[Partner] ) <> ""
							&& Policy[Partner] <> "__BLANK__"
					)
				RETURN
					IF (
						ISBLANK ( _PeriodStart ) || ISBLANK ( _PeriodEnd ) || _PeriodStart > _PeriodEnd,
						BLANK (),
						COALESCE (
							SUMX (
								_PartnerValues,
								VAR _CurrentPartner = Policy[Partner]
								VAR _RowsForPartner =
									CALCULATE (
										[Policy Row Count],
										REMOVEFILTERS ( 'Date' ),
										REMOVEFILTERS ( 'Date Filter' ),
										REMOVEFILTERS ( Policy[Policy Created Date] ),
										FILTER (
											_PolicyRows,
											Policy[Partner] = _CurrentPartner
												&& Policy[Policy Created Date] >= _PeriodStart
												&& Policy[Policy Created Date] <= _PeriodEnd
										)
									)
								RETURN IF ( _RowsForPartner > 0, 1, 0 )
							),
							0
						)
					)
			formatString: #,##0
			displayFolder: Sales

		measure 'Active Policies Last 12M' =
				VAR _AxisMonths = VALUES ( 'Monthly Trend Axis'[MonthStart] )
				VAR _PlanRollup = SELECTEDVALUE ( Sales[SC+ Plan Rollup] )
				VAR _IsShownRollup =
					NOT ISINSCOPE ( Sales[SC+ Plan Rollup] )
						|| _PlanRollup IN { "2YR", "1YR", "Monthly" }
				RETURN
					IF (
						_IsShownRollup,
						CALCULATE (
							[Active Policies],
							REMOVEFILTERS ( 'Date' ),
							REMOVEFILTERS ( 'Date Filter' ),
							TREATAS ( _AxisMonths, 'Date'[StartOfMonth] )
						)
					)
			formatString: #,##0
			displayFolder: Sales\Guide

		measure 'Monthly Paid Revenue Units Last 12M' =
				VAR _AxisMonths = VALUES ( 'Monthly Trend Axis'[MonthStart] )
				RETURN
					CALCULATE (
						[Active Monthly Subscribers],
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						TREATAS ( _AxisMonths, 'Date'[StartOfMonth] )
					)
			formatString: #,##0
			displayFolder: SC+ Plan Classification

		measure 'Policy KPI Validation' =
				"Rows=" & FORMAT ( CALCULATE ( COUNTROWS ( Policy ), REMOVEFILTERS ( Policy ), REMOVEFILTERS ( 'Date' ), REMOVEFILTERS ( 'Date Filter' ) ), "0" )
					& " | Start=" & FORMAT ( [Policy Data Start Date], "yyyy-mm-dd" )
					& " | End=" & FORMAT ( [Policy Data End Date], "yyyy-mm-dd" )
					& " | Active=" & FORMAT ( [Active Policies], "0" )
					& " | Monthly=" & FORMAT ( [Active Monthly Subscribers], "0" )
					& " | Cancels=" & FORMAT ( [Paid Monthly Cancellation Events], "0" )
					& " | Exposure=" & FORMAT ( [Paid Monthly Opening Exposure], "0" )
					& " | Rate=" & FORMAT ( [Monthly Cancellation Rate], "0.0000%" )
			displayFolder: Policy Diagnostics
