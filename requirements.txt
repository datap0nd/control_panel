					            RawText = if Value = null then null else Text.Trim(Text.From(Value)),
					            Invalid = RawText = null or RawText = "" or Text.Upper(RawText) = "NULL" or Text.Contains(RawText, "#"),
					            Parsed =
					                if Invalid then null

								otherwise null
					        in
					            Parsed,
					    ParseMonthKey = (Value as any) as nullable date =>
					        let
					            RawText = if Value = null then null else Text.Trim(Text.From(Value)),
					            MonthText = if RawText = null or Text.Length(RawText) < 7 then null else Text.Start(RawText, 7) & "-01",
					            Parsed = if MonthText = null then null else try Date.FromText(MonthText, [Format="yyyy-MM-dd", Culture="en-US"]) otherwise null
					        in
					            Parsed,
					    ParseNumber = (Value as any) as nullable number =>
					        let
					            RawText = if Value = null then null else Text.Trim(Text.From(Value)),
					            CleanText = if RawText = null then null else Text.Replace(RawText, ",", ""),
					            Parsed =
					                if CleanText = null or CleanText = "" or Text.Upper(CleanText) = "NULL" or Text.Contains(CleanText, "#") then null
								else try Number.FromText(CleanText, "en-US")
								otherwise try Number.From(Value)
					                otherwise null
					        in
					            Parsed,
					    ParseWholeNumber = (Value as any) as nullable number =>
					        let
					            Parsed = ParseNumber(Value)
					        in
					            if Parsed = null then null else Int64.From(Parsed),
					    DataFolder = Text.Trim(DataPath),
					    Backslash = Character.FromNumber(92),
					    PathSeparator = if Text.EndsWith(DataFolder, Backslash) or Text.EndsWith(DataFolder, "/") then "" else if Text.Contains(DataFolder, Backslash) then Backslash else "/",
					    PolicyFilePath = DataFolder & PathSeparator & PolicyCsvFileName,
					    CsvBinary = File.Contents(PolicyFilePath),

					        "product_purchase_date", "warranty_start_date", "warranty_end_date", "cancellation_date",
					        "cancellation_amount", "product_monthly_amount", "product_sale_type", "sale_type",
					        "service_deleted_flag", "deleted_flag", "load_timestamp", "policy_source_rows",
					        "duplicate_policy_id_flag", "cancelled_by_date_flag", "warranty_expired_by_date_flag",
					        "reference_perks_category", "reference_service_product_flag", "reference_device_model_name",
					        "product_reference_rows", "active_product_reference_rows", "product_reference_ambiguous_flag"
					    }, MissingField.UseNull),

					    #"Parsed Date Columns" = Table.TransformColumns(#"Renamed Parsed Policy Created Date", {

					        {"warranty_start_date", ParseDate, type date},
					        {"warranty_end_date", ParseDate, type date},
					        {"cancellation_date", ParseDate, type date}
					    }, null, MissingField.UseNull),
					    #"Parsed Numeric Columns" = Table.TransformColumns(#"Parsed Date Columns", {
					        {"cancellation_amount", ParseNumber, type number},
					        {"product_monthly_amount", ParseNumber, type number},
					        {"policy_source_rows", ParseWholeNumber, Int64.Type},
					        {"duplicate_policy_id_flag", ParseWholeNumber, Int64.Type},
					        {"cancelled_by_date_flag", ParseWholeNumber, Int64.Type},
					        {"warranty_expired_by_date_flag", ParseWholeNumber, Int64.Type},
					        {"product_reference_rows", ParseWholeNumber, Int64.Type},
					        {"active_product_reference_rows", ParseWholeNumber, Int64.Type},
					        {"product_reference_ambiguous_flag", ParseWholeNumber, Int64.Type}
