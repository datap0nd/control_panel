EVALUATE
VAR _PnlCompanies =
    SELECTCOLUMNS (
        FILTER ( VALUES ( 'ERP P&L'[Company Key] ), NOT ISBLANK ( 'ERP P&L'[Company Key] ) ),
        "Company Key", 'ERP P&L'[Company Key]
    )
VAR _MasterCompanies =
    SELECTCOLUMNS (
        FILTER ( VALUES ( 'Subsidiary Master'[Subsidiary] ), NOT ISBLANK ( 'Subsidiary Master'[Subsidiary] ) ),
        "Company Key", UPPER ( TRIM ( 'Subsidiary Master'[Subsidiary] ) )
    )
RETURN
    EXCEPT ( _PnlCompanies, _MasterCompanies )
ORDER BY
    [Company Key]
