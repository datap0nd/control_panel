/// Paste-ready TMDL for aggregated_policy.csv.
/// Apply after exporting sql/aggregated_policy.sql to aggregated_policy.csv.
/// This creates a separate aggregate fact table and does not replace Policy.

createOrReplace

	table 'Aggregated Policy'

		column 'Policy Created Date'
			dataType: dateTime
			formatString: yyyy-mm-dd
			sourceColumn: policy_created_date

			annotation UnderlyingDateTimeDataType = Date

		column 'Policy Created Month Key'
			dataType: string
			isHidden
			sourceColumn: policy_created_month_key

		column 'Country Code'
			dataType: string
			sourceColumn: country_code

		column Region
			dataType: string
			sourceColumn: region

		column 'Sales Channel Code'
			dataType: string
			sourceColumn: sales_channel_code

		column 'Sales Channel'
			dataType: string
			sourceColumn: sales_channel_name

		column 'Partner Name'
			dataType: string
			sourceColumn: partner_name

		column 'New Policy Count'
			dataType: int64
			isHidden
			formatString: #,##0
			summarizeBy: sum
			sourceColumn: new_policy_count

		column 'Source Policy Rows'
			dataType: int64
			isHidden
			formatString: #,##0
			summarizeBy: sum
			sourceColumn: source_policy_rows

		column 'Duplicate Policy Count'
			dataType: int64
			isHidden
			formatString: #,##0
			summarizeBy: sum
			sourceColumn: duplicate_policy_count

		column 'Source File Path'
			dataType: string
			isHidden
			sourceColumn: source_file_path

		column Subsidiary =
				COALESCE (
					RELATED ( 'Subsidiary Master'[Subsidiary] ),
					"Unmapped"
				)
			dataType: string

		hierarchy 'New Policies Hierarchy'

			level Region
				column: Region

			level Subsidiary
				column: Subsidiary

			level 'Sales Channel'
				column: 'Sales Channel'

			level 'Partner Name'
				column: 'Partner Name'

		partition 'Aggregated Policy' = m
			mode: import
			source =
					let
					    FileName = "aggregated_policy.csv",
					    CandidatePaths = {
					        DataPath & FileName,
					        "C:\Users\r.cunha\Documents\Home\Projects\care+\Data Sources\" & FileName,
					        "//Mac/Home/Documents/care/powerbi/data_sources/" & FileName,
					        "Y:/Documents/care/powerbi/data_sources/" & FileName
					    },
					    CandidateFiles = List.Transform(
					        CandidatePaths,
					        (CandidatePath) => [
					            Path = CandidatePath,
					            Content = try File.Contents(CandidatePath) otherwise null
					        ]
					    ),
					    ExistingFiles = List.Select(CandidateFiles, each [Content] <> null),
					    SelectedFile =
					        if List.Count(ExistingFiles) > 0 then ExistingFiles{0}
					        else error "Could not load " & FileName,
					    Source = Csv.Document(
					        SelectedFile[Content],
					        [Delimiter=",", Encoding=65001, QuoteStyle=QuoteStyle.Csv]
					    ),
					    #"Removed Numbers Title Row" =
					        if (try Text.StartsWith(Text.From(Source{0}[Column1]), "Table ") otherwise false)
					        then Table.Skip(Source, 1)
					        else Source,
					    #"Promoted Headers" = Table.PromoteHeaders(
					        #"Removed Numbers Title Row",
					        [PromoteAllScalars=true]
					    ),
					    RequiredColumns = {
					        "policy_created_date",
					        "policy_created_month_key",
					        "country_code",
					        "region",
					        "sales_channel_code",
					        "sales_channel_name",
					        "partner_name",
					        "new_policy_count",
					        "source_policy_rows",
					        "duplicate_policy_count"
					    },
					    MissingRequiredColumns = List.Difference(
					        RequiredColumns,
					        Table.ColumnNames(#"Promoted Headers")
					    ),
					    #"Validated Schema" =
					        if List.Count(MissingRequiredColumns) = 0 then #"Promoted Headers"
					        else error "Aggregated Policy schema mismatch. Missing: " & Text.Combine(MissingRequiredColumns, ", "),
					    #"Changed Type" = Table.TransformColumnTypes(
					        #"Validated Schema",
					        {
					            {"policy_created_date", type date},
					            {"policy_created_month_key", type text},
					            {"country_code", type text},
					            {"region", type text},
					            {"sales_channel_code", type text},
					            {"sales_channel_name", type text},
					            {"partner_name", type text},
					            {"new_policy_count", Int64.Type},
					            {"source_policy_rows", Int64.Type},
					            {"duplicate_policy_count", Int64.Type}
					        },
					        [Culture="en-US", MissingField=MissingField.Error]
					    ),
					    #"Added Source File Path" = Table.AddColumn(
					        #"Changed Type",
					        "source_file_path",
					        each SelectedFile[Path],
					        type text
					    )
					in
					    #"Added Source File Path"

	ref table Measures_table

		measure 'New Policies' = SUM ( 'Aggregated Policy'[New Policy Count] )
			formatString: #,##0
			displayFolder: New Policies

		measure 'New Policies CY' =
				VAR _Year = [Analysis Year]
				VAR _Months = VALUES ( 'Date'[MonthNum] )
				RETURN
					CALCULATE (
						[New Policies],
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						TREATAS ( _Months, 'Date'[MonthNum] ),
						'Date'[Year] = _Year
					)
			formatString: #,##0
			displayFolder: New Policies

		measure 'New Policies LY' =
				VAR _Year = [Analysis Year] - 1
				VAR _Months = VALUES ( 'Date'[MonthNum] )
				RETURN
					CALCULATE (
						[New Policies],
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						TREATAS ( _Months, 'Date'[MonthNum] ),
						'Date'[Year] = _Year
					)
			formatString: #,##0
			displayFolder: New Policies

		measure 'New Policies YoY' =
				DIVIDE (
					[New Policies CY] - [New Policies LY],
					[New Policies LY]
				)
			formatString: +0.0%;-0.0%;0.0%
			displayFolder: New Policies

		measure 'Aggregated Policy Source Rows' = SUM ( 'Aggregated Policy'[Source Policy Rows] )
			formatString: #,##0
			displayFolder: New Policies\Quality

		measure 'Aggregated Policy Duplicate Policies' = SUM ( 'Aggregated Policy'[Duplicate Policy Count] )
			formatString: #,##0
			displayFolder: New Policies\Quality

		fromColumn: 'Aggregated Policy'.'Country Code'
		toColumn: 'Subsidiary Master'.'Country ID'
