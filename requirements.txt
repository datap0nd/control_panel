EVALUATE
SUMMARIZECOLUMNS (
    'ERP P&L'[Source Year],
    'ERP P&L'[Version],
    'ERP P&L'[Month Key],
    "Rows", COUNTROWS ( 'ERP P&L' ),
    "Gross", CALCULATE ( SUM ( 'ERP P&L'[Amount] ), 'ERP P&L'[P&L Line] = "Gross Sales Amt" ),
    "Net", CALCULATE ( SUM ( 'ERP P&L'[Amount] ), 'ERP P&L'[P&L Line] = "Net Sales Amt" ),
    "SalesDeduction", CALCULATE ( SUM ( 'ERP P&L'[Amount] ), 'ERP P&L'[P&L Line] = "Sales Deduction" )
)
ORDER BY
    'ERP P&L'[Source Year],
    'ERP P&L'[Version],
    'ERP P&L'[Month Key]
