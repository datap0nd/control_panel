// Run in Tabular Editor Advanced Scripting while connected to the open model.
// Targets the card label, so the current tooltip wording does not need to match the offline dump.

var heading = "Period-end active policy snapshot";
var body = "Counts policies with active status whose coverage started by the final selected date, warranty has not ended, and cancellation is blank or later than that date. This is a stock at period end, not policies created during the period.";
var pattern = "(<div class=\"\"label\"\">Active Policies</div>.*?<span class=\"\"tooltip\"\">)<strong>.*?</strong><span>.*?</span>";
var replacementCount = 0;
var updatedMeasures = new System.Collections.Generic.List<string>();

foreach (var measure in Model.AllMeasures)
{
    var expression = measure.Expression ?? "";
    var matches = System.Text.RegularExpressions.Regex.Matches(
        expression,
        pattern,
        System.Text.RegularExpressions.RegexOptions.Singleline |
        System.Text.RegularExpressions.RegexOptions.IgnoreCase
    ).Count;

    if (matches == 0)
        continue;

    measure.Expression = System.Text.RegularExpressions.Regex.Replace(
        expression,
        pattern,
        match => match.Groups[1].Value + "<strong>" + heading + "</strong><span>" + body + "</span>",
        System.Text.RegularExpressions.RegexOptions.Singleline |
        System.Text.RegularExpressions.RegexOptions.IgnoreCase
    );
    replacementCount += matches;
    updatedMeasures.Add(measure.Name);
}

if (replacementCount == 0)
    Error("No Active Policies HTML card was found. No model changes were made.");
else
    Info("Updated Active Policies tooltip in " + replacementCount + " card instance(s): " + string.Join(" | ", updatedMeasures));
