	ref table Measures_table

		measure 'Chart Month Context Flag' =
				IF (
					ISINSCOPE ( 'Date'[Month] )
						|| ISINSCOPE ( 'Date'[MonthNum] )
						|| ISINSCOPE ( 'Date'[StartOfMonth] )
						|| ISINSCOPE ( 'Date'[YearMonth] ),
					1,
					0
				)
			formatString: 0
			isHidden
			displayFolder: Helper

		measure 'Chart CY Month Start Date' =
				VAR _Year = [Analysis Year]
				VAR _Month = SELECTEDVALUE ( 'Date'[MonthNum] )
				RETURN
					IF (
						[Chart Month Context Flag] = 1 && NOT ISBLANK ( _Month ),
						DATE ( _Year, _Month, 1 )
					)
			formatString: yyyy-mm-dd
			isHidden
			displayFolder: Helper

		measure 'Chart CY Month End Date' =
				VAR _MonthStart = [Chart CY Month Start Date]
				VAR _DataEnd = [KPI Comparison Data End Date]
				RETURN
					IF (
						ISBLANK ( _MonthStart )
							|| ISBLANK ( _DataEnd )
							|| _MonthStart > _DataEnd,
						BLANK (),
						MIN ( EOMONTH ( _MonthStart, 0 ), _DataEnd )
					)
			formatString: yyyy-mm-dd
			isHidden
			displayFolder: Helper

		measure 'CY Month Scope Sales Volume' =
				VAR _IsChartMonth = [Chart Month Context Flag] = 1
				VAR _PeriodStart =
					IF ( _IsChartMonth, [Chart CY Month Start Date], [KPI Comparison Period Start Date] )
				VAR _PeriodEnd =
					IF ( _IsChartMonth, [Chart CY Month End Date], [KPI Comparison Period End Date] )
				RETURN
					IF (
						ISBLANK ( _PeriodStart ) || ISBLANK ( _PeriodEnd ) || _PeriodStart > _PeriodEnd,
						BLANK (),
						CALCULATE (
							[Sales Volume],
							REMOVEFILTERS ( 'Date' ),
							REMOVEFILTERS ( 'Date Filter' ),
							DATESBETWEEN ( 'Date'[Date], _PeriodStart, _PeriodEnd )
						)
					)
			formatString: #,##0
			isHidden
			displayFolder: Helper
