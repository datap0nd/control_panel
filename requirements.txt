	ref table Measures_table

		measure 'Paid Monthly Raw Cancellation Events' =
				VAR _PeriodStart = [Policy Period Start Date]
				VAR _PeriodEnd = [Policy Period End Date]
				VAR _PolicyRows =
					CALCULATETABLE (
						Policy,
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						REMOVEFILTERS ( Policy[Policy Created Date] ),
						REMOVEFILTERS ( Policy[Warranty Start Date] ),
						REMOVEFILTERS ( Policy[Warranty End Date] ),
						REMOVEFILTERS ( Policy[Cancellation Date] )
					)
				RETURN
					IF (
						ISBLANK ( _PeriodStart )
							|| ISBLANK ( _PeriodEnd )
							|| _PeriodStart > _PeriodEnd,
						BLANK (),
						COUNTROWS (
							FILTER (
								_PolicyRows,
								COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
									&& NOT ISBLANK ( Policy[Cancellation Date] )
									&& Policy[Cancellation Date] >= _PeriodStart
									&& Policy[Cancellation Date] <= _PeriodEnd
							)
						)
					)
			formatString: #,##0
			isHidden
			displayFolder: Helper

		measure 'Paid Monthly Cancellation Events' =
				VAR _PeriodStart = [Policy Period Start Date]
				VAR _PeriodEnd = [Policy Period End Date]
				VAR _PolicyRows =
					CALCULATETABLE (
						Policy,
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						REMOVEFILTERS ( Policy[Policy Created Date] ),
						REMOVEFILTERS ( Policy[Warranty Start Date] ),
						REMOVEFILTERS ( Policy[Warranty End Date] ),
						REMOVEFILTERS ( Policy[Cancellation Date] )
					)
				RETURN
					IF (
						ISBLANK ( _PeriodStart )
							|| ISBLANK ( _PeriodEnd )
							|| _PeriodStart > _PeriodEnd,
						BLANK (),
						COUNTROWS (
							FILTER (
								_PolicyRows,
								VAR _CoverageStart =
									COALESCE (
										Policy[Warranty Start Date],
										Policy[Policy Created Date]
									)
								RETURN
									COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
										&& NOT ISBLANK ( _CoverageStart )
										&& NOT ISBLANK ( Policy[Cancellation Date] )
										&& Policy[Cancellation Date] >= _PeriodStart
										&& Policy[Cancellation Date] <= _PeriodEnd
										&& _CoverageStart <= Policy[Cancellation Date]
										&& (
											ISBLANK ( Policy[Warranty End Date] )
												|| Policy[Warranty End Date] >= Policy[Cancellation Date]
										)
							)
						)
					)
			formatString: #,##0
			isHidden
			displayFolder: Helper

		measure 'Paid Monthly Invalid Cancellation Events' =
				VAR _RawEvents = [Paid Monthly Raw Cancellation Events]
				VAR _ValidEvents = [Paid Monthly Cancellation Events]
				RETURN
					IF (
						ISBLANK ( _RawEvents ),
						BLANK (),
						_RawEvents - _ValidEvents
					)
			formatString: #,##0
			isHidden
			displayFolder: Policy Diagnostics

		measure 'Paid Monthly At-Risk Exposure' =
				VAR _PeriodStart = [Policy Period Start Date]
				VAR _PeriodEnd = [Policy Period End Date]
				VAR _ValidPeriod =
					NOT ISBLANK ( _PeriodStart )
						&& NOT ISBLANK ( _PeriodEnd )
						&& _PeriodStart <= _PeriodEnd
				VAR _SafeStart =
					IF ( _ValidPeriod, _PeriodStart, DATE ( 1900, 1, 1 ) )
				VAR _SafeEnd =
					IF ( _ValidPeriod, _PeriodEnd, _SafeStart )
				VAR _PolicyRows =
					CALCULATETABLE (
						Policy,
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						REMOVEFILTERS ( Policy[Policy Created Date] ),
						REMOVEFILTERS ( Policy[Warranty Start Date] ),
						REMOVEFILTERS ( Policy[Warranty End Date] ),
						REMOVEFILTERS ( Policy[Cancellation Date] )
					)
				VAR _MonthStarts =
					DISTINCT (
						SELECTCOLUMNS (
							CALENDAR ( _SafeStart, _SafeEnd ),
							"MonthStart", DATE ( YEAR ( [Date] ), MONTH ( [Date] ), 1 )
						)
					)
				RETURN
					IF (
						NOT _ValidPeriod,
						BLANK (),
						SUMX (
							_MonthStarts,
							VAR _MonthStart = [MonthStart]
							VAR _SliceStart = MAX ( _PeriodStart, _MonthStart )
							VAR _SliceEnd = MIN ( _PeriodEnd, EOMONTH ( _MonthStart, 0 ) )
							RETURN
								COUNTROWS (
									FILTER (
										_PolicyRows,
										VAR _CoverageStart =
											COALESCE (
												Policy[Warranty Start Date],
												Policy[Policy Created Date]
											)
										RETURN
											COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
												&& NOT ISBLANK ( _CoverageStart )
												&& _CoverageStart <= _SliceEnd
												&& (
													ISBLANK ( Policy[Warranty End Date] )
														|| Policy[Warranty End Date] >= _SliceStart
												)
												&& (
													ISBLANK ( Policy[Cancellation Date] )
														|| Policy[Cancellation Date] >= _SliceStart
												)
									)
								)
						)
					)
			formatString: #,##0
			isHidden
			displayFolder: Helper

		measure 'Paid Monthly Opening Exposure' =
				[Paid Monthly At-Risk Exposure]
			formatString: #,##0
			isHidden
			displayFolder: Helper

		measure 'Monthly Cancellation Rate' =
				DIVIDE (
					[Paid Monthly Cancellation Events],
					[Paid Monthly At-Risk Exposure]
				)
			formatString: 0.0%
			displayFolder: Policy Metrics

		measure 'Monthly Cancellation Rate LY' =
				VAR _DataStart = [Policy Data Start Date]
				VAR _PeriodStart = [Policy Period Start Date]
				VAR _PeriodEnd = [Policy Period End Date]
				VAR _PriorStart = EDATE ( _PeriodStart, -12 )
				VAR _PriorEnd = EDATE ( _PeriodEnd, -12 )
				RETURN
					IF (
						ISBLANK ( _PriorStart )
							|| ISBLANK ( _PriorEnd )
							|| _PriorStart < _DataStart,
						BLANK (),
						CALCULATE (
							[Monthly Cancellation Rate],
							REMOVEFILTERS ( 'Date' ),
							REMOVEFILTERS ( 'Date Filter' ),
							DATESBETWEEN ( 'Date'[Date], _PriorStart, _PriorEnd )
						)
					)
			formatString: 0.0%
			displayFolder: Policy Metrics

		measure 'Policy KPI Validation' =
				"PeriodStart=" & FORMAT ( [Policy Period Start Date], "yyyy-mm-dd" )
					& " | PeriodEnd=" & FORMAT ( [Policy Period End Date], "yyyy-mm-dd" )
					& " | RawEvents=" & FORMAT ( [Paid Monthly Raw Cancellation Events], "0" )
					& " | ValidEvents=" & FORMAT ( [Paid Monthly Cancellation Events], "0" )
					& " | InvalidEvents=" & FORMAT ( [Paid Monthly Invalid Cancellation Events], "0" )
					& " | AtRisk=" & FORMAT ( [Paid Monthly At-Risk Exposure], "0" )
					& " | Rate=" & FORMAT ( [Monthly Cancellation Rate], "0.0000%" )
			displayFolder: Policy Diagnostics
