	ref table Measures_table

		measure 'Month-End Active Monthly Subscribers Last 12M' =
				VAR _AxisMonth = SELECTEDVALUE ( 'Monthly Trend Axis'[MonthStart] )
				VAR _DataEnd = [Policy Data End Date]
				VAR _MonthEnd =
					IF (
						NOT ISBLANK ( _AxisMonth ),
						EOMONTH ( _AxisMonth, 0 )
					)
				VAR _IsLatestMonth =
					NOT ISBLANK ( _AxisMonth )
						&& NOT ISBLANK ( _DataEnd )
						&& YEAR ( _AxisMonth ) = YEAR ( _DataEnd )
						&& MONTH ( _AxisMonth ) = MONTH ( _DataEnd )
				VAR _HistoricalValue =
					CALCULATE (
						[Active Monthly Subscribers],
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						DATESBETWEEN ( 'Date'[Date], _MonthEnd, _MonthEnd )
					)
				VAR _LatestValue =
					CALCULATE (
						[Active Monthly Subscribers],
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' )
					)
				RETURN
					IF (
						ISBLANK ( _AxisMonth )
							|| ISBLANK ( _DataEnd )
							|| _AxisMonth > _DataEnd,
						BLANK (),
						IF (
							_IsLatestMonth,
							COALESCE ( _LatestValue, 0 ),
							COALESCE ( _HistoricalValue, 0 )
						)
					)
			formatString: #,##0
			displayFolder: SC+ Plan Classification
