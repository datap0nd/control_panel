	ref table Measures_table

		measure 'Active Policies Last 12M' =
				VAR _AxisMonth = SELECTEDVALUE ( 'Monthly Trend Axis'[MonthStart] )
				VAR _DataEnd = [Policy Data End Date]
				VAR _SnapshotDate =
					IF (
						NOT ISBLANK ( _AxisMonth )
							&& NOT ISBLANK ( _DataEnd )
							&& _AxisMonth <= _DataEnd,
						MIN ( EOMONTH ( _AxisMonth, 0 ), _DataEnd )
					)
				VAR _PlanRollup = SELECTEDVALUE ( Sales[SC+ Plan Rollup] )
				VAR _IsShownRollup =
					NOT ISINSCOPE ( Sales[SC+ Plan Rollup] )
						|| _PlanRollup IN { "2YR", "1YR", "Monthly" }
				RETURN
					IF (
						NOT _IsShownRollup || ISBLANK ( _SnapshotDate ),
						BLANK (),
						CALCULATE (
							[Active Policies],
							REMOVEFILTERS ( 'Date' ),
							REMOVEFILTERS ( 'Date Filter' ),
							TREATAS ( { _SnapshotDate }, 'Date'[Date] )
						)
					)
			formatString: #,##0
			displayFolder: Sales\Guide
			lineageTag: 83b01bc6-8ac3-4049-8aab-bee96f32b894
