	ref table Measures_table

		measure 'Attach Sales QTY' = [Attach Sales Volume - Common Grain]
			formatString: #,##0
			displayFolder: Attach Rate\Table

		measure 'Sellout That Stays - Common Grain' = [Attach Sellout Net Units - Common Grain]
			formatString: #,##0
			displayFolder: Attach Rate\Table

		measure 'Total Sellout - Common Grain' = [Attach Sellout Total Units - Common Grain]
			formatString: #,##0
			displayFolder: Attach Rate\Table

		measure 'Raw Attach Rate' = [Attach Rate Total S/O]
			formatString: 0.0%
			displayFolder: Attach Rate

		measure 'Excluding Export Attach Rate' = [Attach Rate Net of Export]
			formatString: 0.0%
			displayFolder: Attach Rate

		measure 'Global Attach Rate CY' =
				CALCULATE (
					[Raw Attach Rate],
					REMOVEFILTERS ( 'Subsidiary Master' ),
					REMOVEFILTERS ( Sales[Attach Subsidiary] )
				)
			formatString: 0.0%
			displayFolder: Attach Rate\Benchmark

		measure 'Global Excluding Export Attach Rate' =
				CALCULATE (
					[Excluding Export Attach Rate],
					REMOVEFILTERS ( 'Subsidiary Master' ),
					REMOVEFILTERS ( Sales[Attach Subsidiary] )
				)
			formatString: 0.0%
			displayFolder: Attach Rate\Benchmark

		measure 'Raw Attach Rate vs Global' =
				[Raw Attach Rate] - [Global Attach Rate CY]
			formatString: +0.0%;-0.0%;0.0%
			displayFolder: Attach Rate\Benchmark

		measure 'Excluding Export Attach Rate vs Global' =
				[Excluding Export Attach Rate] - [Global Excluding Export Attach Rate]
			formatString: +0.0%;-0.0%;0.0%
			displayFolder: Attach Rate\Benchmark

		measure 'Raw vs Global (pts)' =
				( [Raw Attach Rate] - [Global Attach Rate CY] ) * 100
			formatString: +0.0 "pts";-0.0 "pts";0.0 "pts"
			displayFolder: Attach Rate\Table

		measure 'Excl Export vs Global (pts)' =
				( [Excluding Export Attach Rate] - [Global Excluding Export Attach Rate] ) * 100
			formatString: +0.0 "pts";-0.0 "pts";0.0 "pts"
			displayFolder: Attach Rate\Table

		measure 'Excluding Export Attach Rate LY' =
				VAR _CurrentYear = [Analysis Year]
				VAR _PriorYear = _CurrentYear - 1
				VAR _CommonEnd = [Attach Common Data End Date]
				VAR _Months =
					FILTER (
						VALUES ( 'Date'[MonthNum] ),
						_CurrentYear < YEAR ( _CommonEnd )
							|| ( _CurrentYear = YEAR ( _CommonEnd ) && 'Date'[MonthNum] <= MONTH ( _CommonEnd ) )
					)
				RETURN
					CALCULATE (
						[Excluding Export Attach Rate],
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						TREATAS ( _Months, 'Date'[MonthNum] ),
						'Date'[Year] = _PriorYear
					)
			formatString: 0.0%
			displayFolder: Attach Rate\LY

		measure 'Excluding Export Attach Rate vs LY' =
				DIVIDE (
					[Excluding Export Attach Rate] - [Excluding Export Attach Rate LY],
					[Excluding Export Attach Rate LY]
				)
			formatString: +0.0%;-0.0%;0.0%
			displayFolder: Attach Rate\LY

		measure 'Attach Table Validation' =
				"Sales=" & FORMAT ( [Attach Sales QTY], "#,##0" )
					& " | Total SO=" & FORMAT ( [Total Sellout - Common Grain], "#,##0" )
					& " | Raw=" & FORMAT ( [Raw Attach Rate], "0.000%" )
					& " | Net SO=" & FORMAT ( [Sellout That Stays - Common Grain], "#,##0" )
					& " | Excl=" & FORMAT ( [Excluding Export Attach Rate], "0.000%" )
			displayFolder: Attach Rate\Diagnostics
