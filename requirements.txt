EVALUATE
VAR _Result =
    SUMMARIZECOLUMNS (
        'P&L'[Date],
        'P&L'[Combined Company],
        FILTER (
            ALL ( 'P&L'[Date] ),
            'P&L'[Date] >= DATE ( 2025, 1, 1 )
                && 'P&L'[Date] < DATE ( 2027, 1, 1 )
        ),
        "Actual Gross Sales",
            CALCULATE (
                SUM ( 'P&L'[Value] ),
                'P&L'[Individual] IN { "CY", "LY" },
                'P&L'[Sales Consolidated Detail (P2)] = "Gross Sales Amt"
            ),
        "Actual Net Sales",
            CALCULATE (
                SUM ( 'P&L'[Value] ),
                'P&L'[Individual] IN { "CY", "LY" },
                'P&L'[Sales Consolidated Detail (P2)] = "Net Sales Amt"
            ),
        "Actual Sales Deduction",
            CALCULATE (
                SUM ( 'P&L'[Value] ),
                'P&L'[Individual] IN { "CY", "LY" },
                'P&L'[Sales Consolidated Detail (P2)] = "Sales Deduction"
            ),
        "Actual Gross Check",
            CALCULATE (
                SUM ( 'P&L'[Value] ),
                'P&L'[Individual] IN { "CY", "LY" },
                'P&L'[Sales Consolidated Detail (P2)] = "Net Sales Amt"
            )
                + CALCULATE (
                    SUM ( 'P&L'[Value] ),
                    'P&L'[Individual] IN { "CY", "LY" },
                    'P&L'[Sales Consolidated Detail (P2)] = "Sales Deduction"
                ),
        "Plan Gross Sales",
            CALCULATE (
                SUM ( 'P&L'[Value] ),
                'P&L'[Individual] = "Plan",
                'P&L'[Sales Consolidated Detail (P2)] = "Gross Sales Amt"
            ),
        "Plan Net Sales",
            CALCULATE (
                SUM ( 'P&L'[Value] ),
                'P&L'[Individual] = "Plan",
                'P&L'[Sales Consolidated Detail (P2)] = "Net Sales Amt"
            ),
        "Fact Rows", CALCULATE ( COUNTROWS ( 'P&L' ) )
    )
RETURN
    FILTER (
        _Result,
        NOT ISBLANK ( [Actual Gross Sales] )
            || NOT ISBLANK ( [Actual Net Sales] )
            || NOT ISBLANK ( [Plan Gross Sales] )
            || NOT ISBLANK ( [Plan Net Sales] )
    )
ORDER BY
    'P&L'[Date],
    'P&L'[Combined Company]
