-- Aggregated new-policy extract for the Power BI sales hierarchy matrix.
--
-- Output grain:
--   policy_created_date
--   country_code
--   region
--   sales_channel_code / sales_channel_name
--   partner_name
--
-- Metric:
--   new_policy_count = distinct CP_SRL_NUM after latest-row deduplication.
--
-- Privacy contract:
--   CP_SRL_NUM is used only inside BigQuery to deduplicate and count. No
--   policy, insurance-policy, user, device, or partner identifier is exported.
--
-- Export the result as aggregated_policy.csv. Keep at least the current and
-- previous calendar year so the Power BI YoY measures have comparable data.

WITH params AS (
  SELECT
    DATE_SUB(DATE_TRUNC(CURRENT_DATE(), YEAR), INTERVAL 1 YEAR) AS start_date,
    CURRENT_DATE() AS end_date
),

-- Add only approved 1:1 code-to-name mappings here. The empty seed keeps the
-- query valid before mappings are supplied. Example:
--   UNION ALL SELECT '1', 'Retail'
channel_map AS (
  SELECT
    CAST(NULL AS STRING) AS sales_channel_code,
    CAST(NULL AS STRING) AS sales_channel_name
  WHERE FALSE
),

policy_normalized AS (
  SELECT
    NULLIF(TRIM(CAST(CP_SRL_NUM AS STRING)), '') AS policy_key_for_aggregation,
    COALESCE(
      SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS DATE),
      DATE(SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS TIMESTAMP)),
      SAFE.PARSE_DATE(
        '%Y%m%d',
        REGEXP_REPLACE(CAST(INSR_PLCY_CRT_DT AS STRING), r'[^0-9]', '')
      )
    ) AS policy_created_date,
    NULLIF(TRIM(CAST(cnty_cd_AM_cntry AS STRING)), '') AS country_code,
    NULLIF(TRIM(CAST(SALE_CHNL_CD AS STRING)), '') AS sales_channel_code,
    NULLIF(TRIM(CAST(PRTNR_NM AS STRING)), '') AS partner_name,
    SAFE_CAST(CAST(sdw_ld_dt AS STRING) AS TIMESTAMP) AS load_timestamp
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_PRTY_SVC_ec1_cntry_AM`
),

policy_ranked AS (
  SELECT
    *,
    COUNT(*) OVER (PARTITION BY policy_key_for_aggregation) AS source_policy_rows,
    ROW_NUMBER() OVER (
      PARTITION BY policy_key_for_aggregation
      ORDER BY
        load_timestamp DESC NULLS LAST,
        policy_created_date DESC NULLS LAST,
        sales_channel_code DESC NULLS LAST,
        partner_name DESC NULLS LAST
    ) AS policy_row_rank
  FROM policy_normalized
  WHERE policy_key_for_aggregation IS NOT NULL
),

policy_deduped AS (
  SELECT
    policy_created_date,
    country_code,
    sales_channel_code,
    partner_name,
    source_policy_rows
  FROM policy_ranked
  CROSS JOIN params
  WHERE policy_row_rank = 1
    AND policy_created_date BETWEEN params.start_date AND params.end_date
)

SELECT
  p.policy_created_date,
  FORMAT_DATE('%Y-%m', p.policy_created_date) AS policy_created_month_key,
  COALESCE(p.country_code, '__BLANK__') AS country_code,
  CASE
    WHEN p.country_code IS NULL THEN 'Unmapped Region'
    WHEN p.country_code IN ('ARE', 'SAU', 'TUR', 'UAE', 'BHR', 'QAT', 'OMN', 'KWT') THEN 'MENA'
    ELSE 'Other'
  END AS region,
  COALESCE(p.sales_channel_code, '__BLANK__') AS sales_channel_code,
  COALESCE(
    m.sales_channel_name,
    p.sales_channel_code,
    'Unmapped Channel'
  ) AS sales_channel_name,
  COALESCE(p.partner_name, 'Unmapped Partner') AS partner_name,
  COUNT(*) AS new_policy_count,
  SUM(p.source_policy_rows) AS source_policy_rows,
  COUNTIF(p.source_policy_rows > 1) AS duplicate_policy_count
FROM policy_deduped AS p
LEFT JOIN channel_map AS m
  ON m.sales_channel_code = p.sales_channel_code
GROUP BY 1, 2, 3, 4, 5, 6, 7
ORDER BY 1, 3, 5, 7;
