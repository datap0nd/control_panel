	ref table Measures_table

		measure 'Policy Period Start Date' =
				VAR _DataStart = [Policy Data Start Date]
				VAR _HasDateContext = [Policy Date Context Flag] = 1
				VAR _SelectedStart = MIN ( 'Date'[Date] )
				RETURN
					IF (
						ISBLANK ( _DataStart ),
						BLANK (),
						IF ( _HasDateContext, _SelectedStart, _DataStart )
					)
			formatString: yyyy-mm-dd
			isHidden
			displayFolder: Helper

		measure 'Policy Period End Date' =
				VAR _DataEnd = [Policy Data End Date]
				VAR _HasDateContext = [Policy Date Context Flag] = 1
				VAR _SelectedEnd = MAX ( 'Date'[Date] )
				RETURN
					IF (
						ISBLANK ( _DataEnd ),
						BLANK (),
						IF ( _HasDateContext, MIN ( _SelectedEnd, _DataEnd ), _DataEnd )
					)
			formatString: yyyy-mm-dd
			isHidden
			displayFolder: Helper
