createOrReplace
	ref table Measures_table
		measure 'Paid Monthly Cancellation Events' =
				VAR _PeriodStart = [Policy Period Start Date]
				VAR _PeriodEnd = [Policy Period End Date]
				VAR _CountryCodes = VALUES ( Sales[Country Code] )
				VAR _CountryFilterActive =
					ISFILTERED ( Sales[Region] )
						|| ISFILTERED ( Sales[Country Code] )
						|| ISFILTERED ( Sales[Subsidiary] )
				VAR _SelectedSalesChannels = VALUES ( Sales[Sales Channel] )
				VAR _SalesChannelCodes =
					SELECTCOLUMNS (
						_SelectedSalesChannels,
						"__ChannelCode",
						SWITCH (
							UPPER ( TRIM ( Sales[Sales Channel] ) ),
							"RETAIL", "5",
							"SES", "53",
							"OPERATOR", "4",
							"E-STORE", "61",
							"ESTORE", "61",
							"OTHERS", "104",
							"B2B", "47",
							Sales[Sales Channel]
						)
					)
				VAR _ProductTypes = VALUES ( Sales[Product Type] )
				VAR _DeviceModels = VALUES ( Sales[Device Model] )
				VAR _PaymentTypes = VALUES ( Sales[Payment Type] )
				VAR _PlanRollups = VALUES ( Sales[SC+ Plan Rollup] )
				VAR _PlanTypes = VALUES ( Sales[SC+ Plan Type] )
				VAR _Partners = VALUES ( Sales[Account / Partner] )
				VAR _DeviceFilterActive =
					ISFILTERED ( Sales[Product Family] )
						|| ISFILTERED ( Sales[Device Series] )
						|| ISFILTERED ( Sales[Device Category] )
						|| ISFILTERED ( Sales[Device Model] )
				VAR _BasePolicyRows =
					CALCULATETABLE (
						SELECTCOLUMNS (
							Policy,
							"__CreatedDate", Policy[Policy Created Date],
							"__WarrantyStart", Policy[Warranty Start Date],
							"__WarrantyEnd", Policy[Warranty End Date],
							"__CancellationDate", Policy[Cancellation Date],
							"__MonthlyAmount", Policy[Product Monthly Amount],
							"__CountryCode", Policy[Country Code],
							"__SalesChannelCode", Policy[Sales Channel Code],
							"__ProductType", Policy[Reference Product Type],
							"__DeviceModel", Policy[Device Model],
							"__PaymentType", Policy[Payment Type],
							"__PlanRollup", Policy[SC+ Plan Rollup],
							"__PlanType", Policy[SC+ Plan Type],
							"__Partner", Policy[Partner]
						),
						REMOVEFILTERS ( 'Date' ),
						REMOVEFILTERS ( 'Date Filter' ),
						REMOVEFILTERS ( Policy[Policy Created Date] ),
						REMOVEFILTERS ( Policy[Warranty Start Date] ),
						REMOVEFILTERS ( Policy[Warranty End Date] ),
						REMOVEFILTERS ( Policy[Cancellation Date] )
					)
				VAR _ScopedPolicyRows =
					FILTER (
						_BasePolicyRows,
						( NOT _CountryFilterActive || [__CountryCode] IN _CountryCodes )
							&& ( NOT ISFILTERED ( Sales[Sales Channel] ) || [__SalesChannelCode] IN _SalesChannelCodes )
							&& ( NOT ISFILTERED ( Sales[Product Type] ) || [__ProductType] IN _ProductTypes )
							&& ( NOT _DeviceFilterActive || [__DeviceModel] IN _DeviceModels )
							&& ( NOT ISFILTERED ( Sales[Payment Type] ) || [__PaymentType] IN _PaymentTypes )
							&& ( NOT ISFILTERED ( Sales[SC+ Plan Rollup] ) || [__PlanRollup] IN _PlanRollups )
							&& ( NOT ISFILTERED ( Sales[SC+ Plan Type] ) || [__PlanType] IN _PlanTypes )
							&& ( NOT ISFILTERED ( Sales[Account / Partner] ) || [__Partner] IN _Partners )
					)
				RETURN
					IF (
						ISBLANK ( _PeriodStart )
							|| ISBLANK ( _PeriodEnd )
							|| _PeriodStart > _PeriodEnd,
						BLANK (),
						COUNTROWS (
							FILTER (
								_ScopedPolicyRows,
								COALESCE ( [__MonthlyAmount], 0 ) > 0
									&& NOT ISBLANK ( [__CancellationDate] )
									&& [__CancellationDate] >= _PeriodStart
									&& [__CancellationDate] <= _PeriodEnd
							)
						)
					)
			formatString: #,##0
			isHidden
			displayFolder: Helper
		measure 'Paid Monthly Opening Exposure' =
				VAR _PeriodStart = [Policy Period Start Date]
				VAR _PeriodEnd = [Policy Period End Date]
