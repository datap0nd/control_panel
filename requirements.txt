var updatedMeasures = new System.Collections.Generic.HashSet<string>();
var replacementCount = 0;
var missingDefinitions = new System.Collections.Generic.List<string>();

foreach (var definition in definitions)
{
    var pattern = "<strong>" + definition.Pattern + "</strong><span>.*?</span>";
    var replacement = "<strong>" + definition.Heading + "</strong><span>" + definition.Body + "</span>";
    var definitionHits = 0;

    foreach (var measure in Model.AllMeasures)
    {
        var expression = measure.Expression ?? "";
        var matches = System.Text.RegularExpressions.Regex.Matches(
            expression,
            pattern,
            System.Text.RegularExpressions.RegexOptions.Singleline
        ).Count;

        if (matches == 0)
            continue;

        measure.Expression = System.Text.RegularExpressions.Regex.Replace(
            expression,
            pattern,
            replacement,
            System.Text.RegularExpressions.RegexOptions.Singleline
        );
        definitionHits += matches;
        replacementCount += matches;
        updatedMeasures.Add(measure.Name);
    }

    if (definitionHits == 0)
        missingDefinitions.Add(definition.Heading);
}

var remainingEmbeddedTooltips = 0;
foreach (var measure in Model.AllMeasures)
{
    remainingEmbeddedTooltips += System.Text.RegularExpressions.Regex.Matches(
        measure.Expression ?? "",
        @"<span class=""""tooltip"""">",
        System.Text.RegularExpressions.RegexOptions.Singleline
    ).Count;
}

var message =
    "Updated measures: " + updatedMeasures.Count +
    "\nTooltip replacements: " + replacementCount +
    "\nEmbedded tooltip instances after update: " + remainingEmbeddedTooltips;

if (missingDefinitions.Count > 0)
    message += "\nDefinitions not found in this model: " + string.Join(" | ", missingDefinitions);

Info(message);
