DEFINE
    VAR _SalesBase =
        FILTER (
            SELECTCOLUMNS (
                ALL ( Sales ),
                "Month Key", Sales[Month Key],
                "Join Sub",
                    COALESCE (
                        RELATED ( 'Subsidiary Master'[Subsidiary] ),
                        Sales[Subsidiary]
                    ),
                "Normalized Channel",
                    SUBSTITUTE (
                        SUBSTITUTE (
                            UPPER ( TRIM ( COALESCE ( Sales[Sales Channel], "" ) ) ),
                            "-",
                            ""
                        ),
                        " ",
                        ""
                    ),
                "Sales Units", COALESCE ( Sales[sales_volume], 0 )
            ),
            NOT ISBLANK ( [Month Key] )
                && [Join Sub] IN { "SGE", "SESAR", "SETK" }
                && CONTAINSSTRING ( [Normalized Channel], "ESTORE" )
        )

    VAR _SalesByMonthSub =
        GROUPBY (
            _SalesBase,
            [Month Key],
            [Join Sub],
            "Sales eStore Units", SUMX ( CURRENTGROUP (), [Sales Units] )
        )

    VAR _SelloutBase =
        FILTER (
            SELECTCOLUMNS (
                ALL ( 'Sellout Stays' ),
                "Month Key", 'Sellout Stays'[Month Key],
                "Join Sub",
                    COALESCE (
                        RELATED ( 'Subsidiary Master'[Subsidiary] ),
                        'Sellout Stays'[Subsidiary]
                    ),
                "Sales Channel", COALESCE ( 'Sellout Stays'[Sales Channel], "" ),
                "Sub Sales Channel", COALESCE ( 'Sellout Stays'[sub_sales_channel_name], "" ),
                "Sell-in AP1", COALESCE ( 'Sellout Stays'[Sell-in AP1], "" ),
                "Account / Partner", COALESCE ( 'Sellout Stays'[Account / Partner], "" ),
                "Sell-in Account", COALESCE ( 'Sellout Stays'[Sell-in Account], "" ),
                "Stays Units", COALESCE ( 'Sellout Stays'[so_that_stays_units], 0 ),
                "Total Units", COALESCE ( 'Sellout Stays'[active_units_resolved], 0 )
            ),
            NOT ISBLANK ( [Month Key] )
                && [Join Sub] IN { "SGE", "SESAR", "SETK" }
        )

    VAR _CandidateRows =
        UNION (
            SELECTCOLUMNS (
                FILTER (
                    _SelloutBase,
                    CONTAINSSTRING (
                        SUBSTITUTE ( SUBSTITUTE ( UPPER ( TRIM ( [Sales Channel] ) ), "-", "" ), " ", "" ),
                        "ESTORE"
                    )
                ),
                "Month Key", [Month Key],
                "Join Sub", [Join Sub],
                "Candidate Field", "Sales Channel",
                "Candidate Value", [Sales Channel],
                "Stays Units", [Stays Units],
                "Total Units", [Total Units]
            ),
            SELECTCOLUMNS (
                FILTER (
                    _SelloutBase,
                    CONTAINSSTRING (
                        SUBSTITUTE ( SUBSTITUTE ( UPPER ( TRIM ( [Sub Sales Channel] ) ), "-", "" ), " ", "" ),
                        "ESTORE"
                    )
                ),
                "Month Key", [Month Key],
                "Join Sub", [Join Sub],
                "Candidate Field", "Sub Sales Channel",
                "Candidate Value", [Sub Sales Channel],
                "Stays Units", [Stays Units],
                "Total Units", [Total Units]
            ),
            SELECTCOLUMNS (
                FILTER (
                    _SelloutBase,
                    CONTAINSSTRING (
                        SUBSTITUTE ( SUBSTITUTE ( UPPER ( TRIM ( [Sell-in AP1] ) ), "-", "" ), " ", "" ),
                        "ESTORE"
                    )
                ),
                "Month Key", [Month Key],
                "Join Sub", [Join Sub],
                "Candidate Field", "Sell-in AP1",
                "Candidate Value", [Sell-in AP1],
                "Stays Units", [Stays Units],
                "Total Units", [Total Units]
            ),
            SELECTCOLUMNS (
                FILTER (
                    _SelloutBase,
                    CONTAINSSTRING (
                        SUBSTITUTE ( SUBSTITUTE ( UPPER ( TRIM ( [Account / Partner] ) ), "-", "" ), " ", "" ),
                        "ESTORE"
                    )
                ),
                "Month Key", [Month Key],
                "Join Sub", [Join Sub],
                "Candidate Field", "Account / Partner",
                "Candidate Value", [Account / Partner],
                "Stays Units", [Stays Units],
                "Total Units", [Total Units]
            ),
            SELECTCOLUMNS (
                FILTER (
                    _SelloutBase,
                    CONTAINSSTRING (
                        SUBSTITUTE ( SUBSTITUTE ( UPPER ( TRIM ( [Sell-in Account] ) ), "-", "" ), " ", "" ),
                        "ESTORE"
                    )
                ),
                "Month Key", [Month Key],
                "Join Sub", [Join Sub],
                "Candidate Field", "Sell-in Account",
                "Candidate Value", [Sell-in Account],
                "Stays Units", [Stays Units],
                "Total Units", [Total Units]
            )
        )

    VAR _CandidateByMonthSub =
        GROUPBY (
            _CandidateRows,
            [Month Key],
            [Join Sub],
            [Candidate Field],
            "Matching Rows", COUNTX ( CURRENTGROUP (), 1 ),
            "eStore Stays Units", SUMX ( CURRENTGROUP (), [Stays Units] ),
            "eStore Total Units", SUMX ( CURRENTGROUP (), [Total Units] )
        )

    VAR _CandidateValues =
        GROUPBY (
            _CandidateRows,
            [Join Sub],
            [Candidate Field],
            [Candidate Value],
            "Matching Rows", COUNTX ( CURRENTGROUP (), 1 ),
            "eStore Stays Units", SUMX ( CURRENTGROUP (), [Stays Units] ),
            "eStore Total Units", SUMX ( CURRENTGROUP (), [Total Units] )
        )

    VAR _MonthlyOutput =
        SELECTCOLUMNS (
            _CandidateByMonthSub,
            "Audit Section", "MONTH + SUB + ESTORE CANDIDATE",
            "Month Key", [Month Key],
            "Join Sub", [Join Sub],
            "Candidate Field", [Candidate Field],
            "Candidate Value", BLANK (),
            "Sales eStore Units",
                VAR _Month = [Month Key]
                VAR _Sub = [Join Sub]
                RETURN
                    MAXX (
                        FILTER ( _SalesByMonthSub, [Month Key] = _Month && [Join Sub] = _Sub ),
                        [Sales eStore Units]
                    ),
            "Matching Rows", [Matching Rows],
            "eStore Stays Units", [eStore Stays Units],
            "eStore Total Units", [eStore Total Units],
            "Attach Rate - Stays",
                VAR _Month = [Month Key]
                VAR _Sub = [Join Sub]
                VAR _Numerator =
                    MAXX (
                        FILTER ( _SalesByMonthSub, [Month Key] = _Month && [Join Sub] = _Sub ),
                        [Sales eStore Units]
                    )
                RETURN DIVIDE ( _Numerator, [eStore Stays Units] ),
            "Attach Rate - Total",
                VAR _Month = [Month Key]
                VAR _Sub = [Join Sub]
                VAR _Numerator =
                    MAXX (
                        FILTER ( _SalesByMonthSub, [Month Key] = _Month && [Join Sub] = _Sub ),
                        [Sales eStore Units]
                    )
                RETURN DIVIDE ( _Numerator, [eStore Total Units] )
        )

    VAR _ValueOutput =
        SELECTCOLUMNS (
            _CandidateValues,
            "Audit Section", "MATCHING ESTORE VALUES",
            "Month Key", BLANK (),
            "Join Sub", [Join Sub],
            "Candidate Field", [Candidate Field],
            "Candidate Value", [Candidate Value],
            "Sales eStore Units", BLANK (),
            "Matching Rows", [Matching Rows],
            "eStore Stays Units", [eStore Stays Units],
            "eStore Total Units", [eStore Total Units],
            "Attach Rate - Stays", BLANK (),
            "Attach Rate - Total", BLANK ()
        )

EVALUATE
    UNION ( _MonthlyOutput, _ValueOutput )

ORDER BY
    [Audit Section],
    [Month Key],
    [Join Sub],
    [Candidate Field],
    [Candidate Value]
