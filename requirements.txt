/// Staged ERP P&L import and diagnostics.
///
/// Work-PC copy with the three source paths taken directly from the
/// (Monthly) MX P&L semantic-model dump.
///
/// The local SC+ list must be DataPath & "erp_master.xlsx" and contain an
/// Excel table named erp_master with one required text column named Material.
///
/// This script intentionally creates no relationships and changes no existing
/// ERP aliases, KPI measures, HTML measures, or report visuals.

createOrReplace

	table 'ERP Master'
		isHidden

		column Material
			dataType: string
			sourceColumn: Material

		column 'Material Key Exact'
			dataType: string
			isHidden
			sourceColumn: Material Key Exact

		column 'Material Key 7'
			dataType: string
			isHidden
			sourceColumn: Material Key 7

		column 'Material Length'
			dataType: int64
			isHidden
			formatString: 0
			summarizeBy: none
			sourceColumn: Material Length

		column 'Source File Path'
			dataType: string
			isHidden
			sourceColumn: Source File Path

		partition 'ERP Master' = m
			mode: import
			source = ```
					let
					    FileName = "erp_master.xlsx",
					    FilePath = DataPath & FileName,
					    NormalizeText = (InputValue as any) as nullable text =>
					        let
					            RawText = try Text.From(InputValue) otherwise null,
					            CleanText = if RawText = null then null else Text.Upper(Text.Trim(Text.Clean(RawText))),
					            Result = if CleanText = "" then null else CleanText
					        in
					            Result,
					    Source = Excel.Workbook(File.Contents(FilePath), null, true),
					    MasterTable =
					        try Source{[Item="erp_master", Kind="Table"]}[Data]
					        otherwise error "ERP Master source must contain an Excel table named erp_master.",
					    RequiredColumns = {"Material"},
					    MissingRequiredColumns = List.Difference(RequiredColumns, Table.ColumnNames(MasterTable)),
					    #"Validated Schema" =
					        if List.Count(MissingRequiredColumns) = 0 then MasterTable
					        else error "ERP Master schema mismatch. Missing: " & Text.Combine(MissingRequiredColumns, ", "),
					    #"Selected Columns" = Table.SelectColumns(#"Validated Schema", RequiredColumns, MissingField.Error),
					    #"Changed Type" = Table.TransformColumnTypes(#"Selected Columns", {{"Material", type text}}, [Culture="en-US", MissingField=MissingField.Error]),
					    #"Added Exact Key" = Table.AddColumn(#"Changed Type", "Material Key Exact", each NormalizeText([Material]), type text),
					    #"Added Seven Character Key" = Table.AddColumn(#"Added Exact Key", "Material Key 7", each if [Material Key Exact] = null then null else Text.Start([Material Key Exact], 7), type text),
					    #"Added Material Length" = Table.AddColumn(#"Added Seven Character Key", "Material Length", each if [Material Key Exact] = null then null else Text.Length([Material Key Exact]), Int64.Type),
					    #"Added Source File Path" = Table.AddColumn(#"Added Material Length", "Source File Path", each FilePath, type text)
					in
					    #"Added Source File Path"
					```

	ref table Measures_table

		measure 'ERP Gross Sales Stage' =
				VAR _Months = VALUES ( 'Date'[YearMonth] )
				VAR _Subsidiaries =
					SELECTCOLUMNS (
						FILTER ( VALUES ( 'Subsidiary Master'[Subsidiary] ), NOT ISBLANK ( 'Subsidiary Master'[Subsidiary] ) ),
						"Company Key", UPPER ( TRIM ( 'Subsidiary Master'[Subsidiary] ) )
					)
				RETURN
					CALCULATE (
						SUM ( 'ERP P&L'[Amount] ),
						KEEPFILTERS ( 'ERP P&L'[P&L Line] = "Gross Sales Amt" ),
						TREATAS ( _Months, 'ERP P&L'[Month Key] ),
						TREATAS ( _Subsidiaries, 'ERP P&L'[Company Key] )
					)
			formatString: #,##0.00
			displayFolder: ERP P&L\Stage

		measure 'ERP Net Sales Stage' =
				VAR _Months = VALUES ( 'Date'[YearMonth] )
				VAR _Subsidiaries =
					SELECTCOLUMNS (
						FILTER ( VALUES ( 'Subsidiary Master'[Subsidiary] ), NOT ISBLANK ( 'Subsidiary Master'[Subsidiary] ) ),
						"Company Key", UPPER ( TRIM ( 'Subsidiary Master'[Subsidiary] ) )
					)
				RETURN
					CALCULATE (
						SUM ( 'ERP P&L'[Amount] ),
						KEEPFILTERS ( 'ERP P&L'[P&L Line] = "Net Sales Amt" ),
						TREATAS ( _Months, 'ERP P&L'[Month Key] ),
						TREATAS ( _Subsidiaries, 'ERP P&L'[Company Key] )
					)
			formatString: #,##0.00
			displayFolder: ERP P&L\Stage

		measure 'ERP Sales Deduction Stage' =
				VAR _Months = VALUES ( 'Date'[YearMonth] )
				VAR _Subsidiaries =
					SELECTCOLUMNS (
						FILTER ( VALUES ( 'Subsidiary Master'[Subsidiary] ), NOT ISBLANK ( 'Subsidiary Master'[Subsidiary] ) ),
						"Company Key", UPPER ( TRIM ( 'Subsidiary Master'[Subsidiary] ) )
					)
				RETURN
					CALCULATE (
						SUM ( 'ERP P&L'[Amount] ),
						KEEPFILTERS ( 'ERP P&L'[P&L Line] = "Sales Deduction" ),
						TREATAS ( _Months, 'ERP P&L'[Month Key] ),
						TREATAS ( _Subsidiaries, 'ERP P&L'[Company Key] )
					)
			formatString: #,##0.00
			displayFolder: ERP P&L\Stage

		measure 'ERP Gross Sales Raw Stage' =
				VAR _Months = VALUES ( 'Date'[YearMonth] )
				VAR _Subsidiaries =
					SELECTCOLUMNS (
						FILTER ( VALUES ( 'Subsidiary Master'[Subsidiary] ), NOT ISBLANK ( 'Subsidiary Master'[Subsidiary] ) ),
						"Company Key", UPPER ( TRIM ( 'Subsidiary Master'[Subsidiary] ) )
					)
				RETURN
					CALCULATE (
						SUM ( 'ERP P&L'[Amount Raw] ),
						KEEPFILTERS ( 'ERP P&L'[P&L Line] = "Gross Sales Amt" ),
						TREATAS ( _Months, 'ERP P&L'[Month Key] ),
						TREATAS ( _Subsidiaries, 'ERP P&L'[Company Key] )
					)
			formatString: #,##0.00
			displayFolder: ERP P&L\Stage\Diagnostics

		measure 'ERP Net Sales Raw Stage' =
				VAR _Months = VALUES ( 'Date'[YearMonth] )
				VAR _Subsidiaries =
					SELECTCOLUMNS (
						FILTER ( VALUES ( 'Subsidiary Master'[Subsidiary] ), NOT ISBLANK ( 'Subsidiary Master'[Subsidiary] ) ),
						"Company Key", UPPER ( TRIM ( 'Subsidiary Master'[Subsidiary] ) )
					)
				RETURN
					CALCULATE (
						SUM ( 'ERP P&L'[Amount Raw] ),
						KEEPFILTERS ( 'ERP P&L'[P&L Line] = "Net Sales Amt" ),
						TREATAS ( _Months, 'ERP P&L'[Month Key] ),
						TREATAS ( _Subsidiaries, 'ERP P&L'[Company Key] )
					)
			formatString: #,##0.00
			displayFolder: ERP P&L\Stage\Diagnostics

		measure 'ERP Gross Formula Delta Stage' =
				[ERP Gross Sales Stage] - ( [ERP Net Sales Stage] + [ERP Sales Deduction Stage] )
			formatString: #,##0.00
			displayFolder: ERP P&L\Stage\Diagnostics

		measure 'ERP Fact Rows Stage' =
				COUNTROWS ( 'ERP P&L' )
			formatString: #,##0
			displayFolder: ERP P&L\Stage\Diagnostics

		measure 'ERP Master Rows Stage' =
				COUNTROWS ( 'ERP Master' )
			formatString: #,##0
			displayFolder: ERP P&L\Stage\Diagnostics

		measure 'ERP Master Blank Material Count Stage' =
				COUNTROWS ( FILTER ( 'ERP Master', ISBLANK ( 'ERP Master'[Material Key Exact] ) ) )
			formatString: #,##0
			displayFolder: ERP P&L\Stage\Diagnostics

		measure 'ERP Master 7 Key Collision Count Stage' =
				VAR _ByKey =
					SUMMARIZE (
						FILTER ( 'ERP Master', NOT ISBLANK ( 'ERP Master'[Material Key 7] ) ),
						'ERP Master'[Material Key 7],
						"ExactKeyCount", DISTINCTCOUNT ( 'ERP Master'[Material Key Exact] )
					)
