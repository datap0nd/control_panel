measure 'Selected Benchmark' =
    VAR _name = SELECTEDVALUE('Metric'[Name], "Net Sales")
    RETURN
        SWITCH(
            _name,
            "Net Sales", [MP Plan Total],
            VAR _peerSubs =
                CALCULATETABLE(
                    ALL(dim_subsidiary[Subsidiary]),
                    REMOVEFILTERS(Subsidiaries[Subsidiary]),
                    REMOVEFILTERS(dim_subsidiary[Subsidiary])
                )
            RETURN
                AVERAGEX(
                    _peerSubs,
                    VAR _peerSub = dim_subsidiary[Subsidiary]
                    RETURN
                        CALCULATE(
                            [Selected Metric],
                            REMOVEFILTERS(Subsidiaries[Subsidiary]),
                            REMOVEFILTERS(dim_subsidiary[Subsidiary]),
                            dim_subsidiary[Subsidiary] = _peerSub
                        )
                )
        )
