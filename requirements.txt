-- Task 5 semantic model and measure validation pack.
-- Dialect: BigQuery GoogleSQL.
--
-- Run sections independently. Sections that use the first validation period
-- use January 2026:
--   start_date = DATE '2026-01-01'
--   end_date   = DATE '2026-01-31'
--
-- This pack does not query sellout because the sellout source is external to
-- BigQuery. Section 09 defines the required output contract for the external
-- source so Task 6 can validate attach rate once sellout is staged or imported.

-- ======================================================================
-- 01. Measure Validation Register
-- ======================================================================

SELECT *
FROM UNNEST([
  STRUCT('sales_volume' AS measure_family, 'Sales Volume = SUM(FactPolicyDailyAggregate[SalesCount])' AS model_expression, '02 official aggregate sales; 03 policy detail parity' AS sql_counterpart, 'Use aggregate as official numerator until policy parity is proven.' AS note),
  STRUCT('active_policies', 'Count policy rows mapped to active status', '04 status and lifecycle profile', 'Blocked until CP_STS_CD mapping is approved.'),
  STRUCT('cancelled_policies', 'SUM(FactPolicyDailyAggregate[CancellationCount]) or approved policy cancellation logic', '04 status and lifecycle profile', 'Aggregate is usable for source-of-truth style counts; policy detail needs mapping.'),
  STRUCT('expired_policies', 'SUM(FactPolicyDailyAggregate[ExpirationCount]) or approved policy expiry logic', '04 status and lifecycle profile', 'Aggregate is usable for source-of-truth style counts; policy detail needs mapping.'),
  STRUCT('gross_revenue', 'SUMX(FactRevenue, RVN_AMT + INSR_PRMUM_AMT + CHNL_AMT + DISTRBTR_AMT)', '06 revenue formula totals; 07 revenue grain quality', 'Formula confirmed by answers.rtf; source fields still need live confirmation.'),
  STRUCT('net_revenue_program_margin', 'SUM(FactRevenue[RVN_AMT])', '06 revenue formula totals; 07 revenue grain quality', 'Formula confirmed by answers.rtf; source fields still need live confirmation.'),
  STRUCT('subsidiary_revenue', 'Paid RVN_TP_CD 11 and 12 plus CNTRBTN_RVN', '06 revenue formula totals; 07 revenue grain quality', 'Formula confirmed by answers.rtf; source fields still need live confirmation.'),
  STRUCT('average_revenue_per_policy', 'DIVIDE([Gross Revenue], [Sales Volume])', '08 revenue to sales shared-grain check', 'Only valid at shared date/country/product/device/channel/payment grain.'),
  STRUCT('sellout_units', 'SUM(FactSellout[SelloutUnits])', '09 external sellout output contract', 'Blocked until external sellout is staged or imported.'),
  STRUCT('attach_rate', 'DIVIDE([Sales Volume], [Sellout Units])', '09 external sellout output contract', 'Raw full denominator; no eligibility filter.'),
  STRUCT('prior_period_prior_year', 'Date-intelligence measures after coverage check', '11 history coverage check', 'Do not publish prior-year measures until history is comparable.'),
  STRUCT('data_quality_flags', 'Counts/flags for missing keys, duplicates, unmapped dimensions, invalid dates, null or negative amounts', '10 data quality facts; 12 model readiness summary', 'Required for Task 6 validation pages.')
])
ORDER BY measure_family;

-- ======================================================================
-- 02. Official Sales, Cancellation, And Expiry Aggregate
-- ======================================================================
-- This is the official reconciled numerator pattern from Task 4. For Task 6,
-- the January 2026 TUR total from this query must match the Power BI measure
-- [Sales Volume] under the same filters.

WITH params AS (
  SELECT DATE '2026-01-01' AS start_date, DATE '2026-01-31' AS end_date
),
aggregate_sales AS (
  SELECT
    COALESCE(
      SAFE_CAST(CAST(STAT_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(STAT_DD AS STRING), r'[^0-9]', ''))
    ) AS sales_date,
    COALESCE(NULLIF(TRIM(CAST(CNTY_CD AS STRING)), ''), '__BLANK__') AS country_code,
    COALESCE(NULLIF(TRIM(CAST(SALE_CHNL_NM AS STRING)), ''), '__BLANK__') AS sale_channel_name,
    COALESCE(NULLIF(TRIM(CAST(SUB_SALE_CHNL_NM AS STRING)), ''), '__BLANK__') AS sub_sale_channel_name,
    COALESCE(NULLIF(TRIM(CAST(PERKS_CTGY AS STRING)), ''), '__BLANK__') AS product_type,
    COALESCE(NULLIF(TRIM(CAST(SVC_PRDCT_YN_FG AS STRING)), ''), '__BLANK__') AS service_product_flag,
    COALESCE(NULLIF(TRIM(CAST(DVCE_CTGY_CD AS STRING)), ''), '__BLANK__') AS device_category_code,
    COALESCE(NULLIF(TRIM(CAST(DVCE_MODEL_NM AS STRING)), ''), '__BLANK__') AS device_model_name,
    COALESCE(NULLIF(TRIM(CAST(SALE_MTHD_CD AS STRING)), ''), '__BLANK__') AS sale_method_code,
    COALESCE(NULLIF(TRIM(CAST(PFM_TP AS STRING)), ''), '__BLANK__') AS performance_type,
    COALESCE(NULLIF(TRIM(CAST(PMT_TP_CD AS STRING)), ''), '__BLANK__') AS payment_type_code,
    SAFE_CAST(SALE_CNT AS NUMERIC) AS sales_count,
    SAFE_CAST(CNCL_CNT AS NUMERIC) AS cancellation_count,
    SAFE_CAST(EPRD_CNT AS NUMERIC) AS expiration_count
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_DSBD_SCP_STAT_CNTRCT_d5f9595b_ec1_cntry_AM`
)
SELECT
  country_code,
  FORMAT_DATE('%Y-%m', sales_date) AS sales_month,
  sale_channel_name,
  sub_sale_channel_name,
  product_type,
  service_product_flag,
  device_category_code,
  device_model_name,
  sale_method_code,
  performance_type,
  payment_type_code,
  SUM(COALESCE(sales_count, 0)) AS sales_volume,
  SUM(COALESCE(cancellation_count, 0)) AS cancelled_policies_aggregate,
  SUM(COALESCE(expiration_count, 0)) AS expired_policies_aggregate,
  COUNT(*) AS source_rows,
  COUNTIF(sales_count IS NULL) AS uncastable_sales_count_rows
FROM aggregate_sales
JOIN params ON TRUE
WHERE sales_date BETWEEN params.start_date AND params.end_date
GROUP BY
  country_code,
  sales_month,
  sale_channel_name,
  sub_sale_channel_name,
  product_type,
  service_product_flag,
  device_category_code,
  device_model_name,
  sale_method_code,
  performance_type,
  payment_type_code
ORDER BY sales_volume DESC;

-- ======================================================================
-- 03. Policy Detail Parity Against Daily Aggregate
-- ======================================================================
-- Proves whether FactPolicySales can be used for [Sales Volume] at shared
-- grain. If this query shows material variance, Power BI should keep aggregate
-- sales as the official sales fact and use policy rows for diagnostics/detail.

WITH params AS (
  SELECT DATE '2026-01-01' AS start_date, DATE '2026-01-31' AS end_date
),
aggregate_sales AS (
  SELECT
    COALESCE(
      SAFE_CAST(CAST(STAT_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(STAT_DD AS STRING), r'[^0-9]', ''))
    ) AS sales_date,
    COALESCE(NULLIF(TRIM(CAST(CNTY_CD AS STRING)), ''), '__BLANK__') AS country_code,
    COALESCE(NULLIF(TRIM(CAST(DVCE_MODEL_NM AS STRING)), ''), '__BLANK__') AS device_model_name,
    COALESCE(NULLIF(TRIM(CAST(PMT_TP_CD AS STRING)), ''), '__BLANK__') AS payment_type_code,
    SUM(COALESCE(SAFE_CAST(SALE_CNT AS NUMERIC), 0)) AS aggregate_sales_count
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_DSBD_SCP_STAT_CNTRCT_d5f9595b_ec1_cntry_AM`
  GROUP BY sales_date, country_code, device_model_name, payment_type_code
),
policy_sales AS (
  SELECT
    COALESCE(
      SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS DATE),
      DATE(SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS TIMESTAMP)),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(INSR_PLCY_CRT_DT AS STRING), r'[^0-9]', ''))
    ) AS policy_date,
    COALESCE(NULLIF(TRIM(CAST(cnty_cd_AM_cntry AS STRING)), ''), '__BLANK__') AS country_code,
    COALESCE(NULLIF(TRIM(CAST(DVCE_MODEL_NM AS STRING)), ''), '__BLANK__') AS device_model_name,
    COALESCE(NULLIF(TRIM(CAST(PMT_TP_CD AS STRING)), ''), '__BLANK__') AS payment_type_code,
    COUNT(DISTINCT NULLIF(TRIM(CAST(CP_SRL_NUM AS STRING)), '')) AS distinct_policy_count,
    COUNT(*) AS policy_source_rows
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_PRTY_SVC_ec1_cntry_AM`
  GROUP BY policy_date, country_code, device_model_name, payment_type_code
)
SELECT
  COALESCE(aggregate_sales.sales_date, policy_sales.policy_date) AS validation_date,
  COALESCE(aggregate_sales.country_code, policy_sales.country_code) AS country_code,
  COALESCE(aggregate_sales.device_model_name, policy_sales.device_model_name) AS device_model_name,
  COALESCE(aggregate_sales.payment_type_code, policy_sales.payment_type_code) AS payment_type_code,
  aggregate_sales.aggregate_sales_count,
  policy_sales.distinct_policy_count,
  COALESCE(aggregate_sales.aggregate_sales_count, 0)
    - COALESCE(policy_sales.distinct_policy_count, 0) AS sales_count_variance,
  policy_sales.policy_source_rows,
  aggregate_sales.sales_date IS NULL AS missing_from_aggregate,
  policy_sales.policy_date IS NULL AS missing_from_policy
FROM aggregate_sales
FULL OUTER JOIN policy_sales
  ON policy_sales.policy_date = aggregate_sales.sales_date
 AND policy_sales.country_code = aggregate_sales.country_code
 AND policy_sales.device_model_name = aggregate_sales.device_model_name
 AND policy_sales.payment_type_code = aggregate_sales.payment_type_code
JOIN params ON TRUE
WHERE COALESCE(aggregate_sales.sales_date, policy_sales.policy_date)
  BETWEEN params.start_date AND params.end_date
ORDER BY
  ABS(COALESCE(aggregate_sales.aggregate_sales_count, 0)
    - COALESCE(policy_sales.distinct_policy_count, 0)) DESC,
  validation_date,
  country_code,
  device_model_name
LIMIT 500;

-- ======================================================================
-- 04. Policy Status And Lifecycle Profile
-- ======================================================================
-- Supports active, cancelled, and expired policy definitions. This is not a
-- final business mapping. It profiles source codes and date availability so a
-- mapping table can be approved before Task 6 publishes these measures.

WITH params AS (
  SELECT DATE '2026-01-01' AS start_date, DATE '2026-01-31' AS end_date
),
policy AS (
  SELECT
    COALESCE(
      SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS DATE),
      DATE(SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS TIMESTAMP)),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(INSR_PLCY_CRT_DT AS STRING), r'[^0-9]', ''))
    ) AS policy_date,
    COALESCE(
      SAFE_CAST(CAST(CNCL_DT AS STRING) AS DATE),
      DATE(SAFE_CAST(CAST(CNCL_DT AS STRING) AS TIMESTAMP)),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(CNCL_DT AS STRING), r'[^0-9]', ''))
    ) AS cancellation_date,
    COALESCE(
      SAFE_CAST(CAST(WRNT_END_DT AS STRING) AS DATE),
      DATE(SAFE_CAST(CAST(WRNT_END_DT AS STRING) AS TIMESTAMP)),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(WRNT_END_DT AS STRING), r'[^0-9]', ''))
    ) AS warranty_end_date,
    COALESCE(NULLIF(TRIM(CAST(cnty_cd_AM_cntry AS STRING)), ''), '__BLANK__') AS country_code,
    COALESCE(NULLIF(TRIM(CAST(CP_STS_CD AS STRING)), ''), '__BLANK__') AS policy_status_code,
    COALESCE(NULLIF(TRIM(CAST(PMT_TP_CD AS STRING)), ''), '__BLANK__') AS payment_type_code,
    COALESCE(NULLIF(TRIM(CAST(LST_PMT_STS_TXT AS STRING)), ''), '__BLANK__') AS last_payment_status_text,
    COALESCE(NULLIF(TRIM(CAST(INSR_CMPNY_TP_CD AS STRING)), ''), '__BLANK__') AS insurer_company_type_code,
    NULLIF(TRIM(CAST(CP_SRL_NUM AS STRING)), '') AS policy_id
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_PRTY_SVC_ec1_cntry_AM`
)
SELECT
  country_code,
  policy_status_code,
  payment_type_code,
  last_payment_status_text,
  insurer_company_type_code,
  COUNT(*) AS source_rows,
  COUNT(DISTINCT policy_id) AS distinct_policies,
  COUNTIF(cancellation_date IS NOT NULL) AS rows_with_cancellation_date,
  COUNTIF(warranty_end_date IS NOT NULL) AS rows_with_warranty_end_date,
  COUNTIF(warranty_end_date < CURRENT_DATE()) AS rows_with_past_warranty_end_date,
  MIN(policy_date) AS min_policy_date,
  MAX(policy_date) AS max_policy_date,
  MIN(cancellation_date) AS min_cancellation_date,
  MAX(cancellation_date) AS max_cancellation_date,
  MIN(warranty_end_date) AS min_warranty_end_date,
  MAX(warranty_end_date) AS max_warranty_end_date
FROM policy
JOIN params ON TRUE
WHERE policy_date BETWEEN params.start_date AND params.end_date
GROUP BY
  country_code,
  policy_status_code,
  payment_type_code,
  last_payment_status_text,
  insurer_company_type_code
ORDER BY distinct_policies DESC, country_code, policy_status_code;

-- ======================================================================
-- 06. Revenue Formula Totals
-- ======================================================================
-- Confirmed formulas from answers.rtf:
-- - Gross revenue = RVN_AMT + INSR_PRMUM_AMT + CHNL_AMT + DISTRBTR_AMT.
-- - Net revenue / program margin = RVN_AMT.
-- - Subsidiary revenue = paid RVN_TP_CD values 11 and 12 plus CNTRBTN_RVN.
--
-- Run only after the live revenue table confirms these columns are present.

WITH params AS (
  SELECT DATE '2026-01-01' AS start_date, DATE '2026-01-31' AS end_date
),
revenue AS (
  SELECT
    COALESCE(
      SAFE_CAST(CAST(RVN_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(RVN_DD AS STRING), r'[^0-9]', ''))
    ) AS revenue_date,
    COALESCE(NULLIF(TRIM(CAST(CNTY_CD AS STRING)), ''), '__BLANK__') AS country_code,
    COALESCE(NULLIF(TRIM(CAST(SALE_CHNL_NM AS STRING)), ''), '__BLANK__') AS sale_channel_name,
    COALESCE(NULLIF(TRIM(CAST(PERKS_CTGY AS STRING)), ''), '__BLANK__') AS product_type,
    COALESCE(NULLIF(TRIM(CAST(SVC_PRDCT_YN_FG AS STRING)), ''), '__BLANK__') AS service_product_flag,
    COALESCE(NULLIF(TRIM(CAST(DVCE_MODEL_NM AS STRING)), ''), '__BLANK__') AS device_model_name,
    COALESCE(NULLIF(TRIM(CAST(PMT_TP_CD AS STRING)), ''), '__BLANK__') AS payment_type_code,
    COALESCE(NULLIF(TRIM(CAST(RVN_TP_CD AS STRING)), ''), '__BLANK__') AS revenue_type_code,
    SAFE_CAST(RVN_AMT AS NUMERIC) AS program_margin_amount,
    SAFE_CAST(INSR_PRMUM_AMT AS NUMERIC) AS insurer_premium_amount,
    SAFE_CAST(CHNL_AMT AS NUMERIC) AS channel_amount,
    SAFE_CAST(DISTRBTR_AMT AS NUMERIC) AS distributor_amount,
    SAFE_CAST(CNTRBTN_RVN AS NUMERIC) AS contribution_revenue,
    SAFE_CAST(RVN_CNT AS NUMERIC) AS revenue_units,
    SAFE_CAST(CNVRN_CNT AS NUMERIC) AS conversion_units
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_DSBD_SCP_RVN_d5f9595b_ec1_cntry_AM`
)
SELECT
  country_code,
  FORMAT_DATE('%Y-%m', revenue_date) AS revenue_month,
  sale_channel_name,
  product_type,
  service_product_flag,
  device_model_name,
  payment_type_code,
  SUM(COALESCE(program_margin_amount, 0)) AS net_revenue_program_margin,
  SUM(COALESCE(insurer_premium_amount, 0)) AS insurer_premium,
  SUM(COALESCE(channel_amount, 0)) AS channel_amount,
  SUM(COALESCE(distributor_amount, 0)) AS distributor_amount,
  SUM(COALESCE(contribution_revenue, 0)) AS contribution_revenue,
  SUM(COALESCE(program_margin_amount, 0)
    + COALESCE(insurer_premium_amount, 0)
    + COALESCE(channel_amount, 0)
    + COALESCE(distributor_amount, 0)) AS gross_revenue,
  SUM(
    CASE
      WHEN revenue_type_code IN ('11', '12') THEN COALESCE(program_margin_amount, 0)
      ELSE 0
    END
    + COALESCE(contribution_revenue, 0)
  ) AS subsidiary_revenue,
  SUM(COALESCE(revenue_units, 0)) AS revenue_units,
  SUM(COALESCE(conversion_units, 0)) AS conversion_units,
  COUNT(*) AS source_rows
FROM revenue
JOIN params ON TRUE
WHERE revenue_date BETWEEN params.start_date AND params.end_date
GROUP BY
  country_code,
  revenue_month,
  sale_channel_name,
  product_type,
  service_product_flag,
  device_model_name,
  payment_type_code
ORDER BY gross_revenue DESC;

-- ======================================================================
-- 07. Revenue Grain And Amount Quality
-- ======================================================================
-- Checks duplicate grain rows, null amount components, and negative amounts.
-- Duplicate rows are not automatically wrong, but they define what FactRevenue
-- grain actually is.

WITH params AS (
  SELECT DATE '2026-01-01' AS start_date, DATE '2026-01-31' AS end_date
),
revenue AS (
  SELECT
    COALESCE(
      SAFE_CAST(CAST(RVN_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(RVN_DD AS STRING), r'[^0-9]', ''))
    ) AS revenue_date,
    COALESCE(NULLIF(TRIM(CAST(CNTY_CD AS STRING)), ''), '__BLANK__') AS country_code,
    COALESCE(NULLIF(TRIM(CAST(SALE_CHNL_NM AS STRING)), ''), '__BLANK__') AS sale_channel_name,
    COALESCE(NULLIF(TRIM(CAST(PERKS_CTGY AS STRING)), ''), '__BLANK__') AS product_type,
    COALESCE(NULLIF(TRIM(CAST(SVC_PRDCT_YN_FG AS STRING)), ''), '__BLANK__') AS service_product_flag,
    COALESCE(NULLIF(TRIM(CAST(DVCE_MODEL_NM AS STRING)), ''), '__BLANK__') AS device_model_name,
    COALESCE(NULLIF(TRIM(CAST(PMT_TP_CD AS STRING)), ''), '__BLANK__') AS payment_type_code,
    COALESCE(NULLIF(TRIM(CAST(RVN_TP_CD AS STRING)), ''), '__BLANK__') AS revenue_type_code,
    SAFE_CAST(RVN_AMT AS NUMERIC) AS program_margin_amount,
    SAFE_CAST(INSR_PRMUM_AMT AS NUMERIC) AS insurer_premium_amount,
    SAFE_CAST(CHNL_AMT AS NUMERIC) AS channel_amount,
    SAFE_CAST(DISTRBTR_AMT AS NUMERIC) AS distributor_amount,
    SAFE_CAST(CNTRBTN_RVN AS NUMERIC) AS contribution_revenue
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_DSBD_SCP_RVN_d5f9595b_ec1_cntry_AM`
),
grain AS (
  SELECT
    revenue_date,
    country_code,
    sale_channel_name,
    product_type,
    service_product_flag,
    device_model_name,
    payment_type_code,
    revenue_type_code,
    COUNT(*) AS source_rows
  FROM revenue
  JOIN params ON TRUE
  WHERE revenue_date BETWEEN params.start_date AND params.end_date
  GROUP BY
    revenue_date,
    country_code,
    sale_channel_name,
    product_type,
    service_product_flag,
    device_model_name,
    payment_type_code,
    revenue_type_code
)
SELECT
  'amount_quality' AS check_name,
  COUNT(*) AS rows_checked,
  COUNTIF(program_margin_amount IS NULL) AS null_program_margin_rows,
  COUNTIF(insurer_premium_amount IS NULL) AS null_insurer_premium_rows,
  COUNTIF(channel_amount IS NULL) AS null_channel_amount_rows,
  COUNTIF(distributor_amount IS NULL) AS null_distributor_amount_rows,
  COUNTIF(contribution_revenue IS NULL) AS null_contribution_revenue_rows,
  COUNTIF(program_margin_amount < 0) AS negative_program_margin_rows,
  COUNTIF(insurer_premium_amount < 0) AS negative_insurer_premium_rows,
  COUNTIF(channel_amount < 0) AS negative_channel_amount_rows,
  COUNTIF(distributor_amount < 0) AS negative_distributor_amount_rows,
  COUNTIF(contribution_revenue < 0) AS negative_contribution_revenue_rows
FROM revenue
JOIN params ON TRUE
WHERE revenue_date BETWEEN params.start_date AND params.end_date

UNION ALL

SELECT
  'duplicate_revenue_grain_rows',
  COUNT(*) AS rows_checked,
  COUNTIF(source_rows > 1) AS duplicate_grain_groups,
  NULL,
  NULL,
  NULL,
  NULL,
  NULL,
  NULL,
  NULL,
  NULL,
  NULL
FROM grain;

-- ======================================================================
-- 08. Revenue To Sales Shared-Grain Check
-- ======================================================================
-- Supports Average Revenue Per Policy and combined sales/revenue reporting.

    )) AS month_key,

      SAFE_CAST(CAST(STAT_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(STAT_DD AS STRING), r'[^0-9]', ''))
    ) BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
  GROUP BY
    month_key,
    country_code,
    sale_channel_name,
    product_type,
    service_product_flag,
    device_model_name,
    payment_type_code
),
revenue AS (
  SELECT
    FORMAT_DATE('%Y-%m', COALESCE(
      SAFE_CAST(CAST(RVN_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(RVN_DD AS STRING), r'[^0-9]', ''))
    )) AS month_key,
    COALESCE(NULLIF(TRIM(CAST(CNTY_CD AS STRING)), ''), '__BLANK__') AS country_code,
    COALESCE(NULLIF(TRIM(CAST(SALE_CHNL_NM AS STRING)), ''), '__BLANK__') AS sale_channel_name,
    COALESCE(NULLIF(TRIM(CAST(PERKS_CTGY AS STRING)), ''), '__BLANK__') AS product_type,

    SUM(COALESCE(SAFE_CAST(RVN_AMT AS NUMERIC), 0)
      + COALESCE(SAFE_CAST(INSR_PRMUM_AMT AS NUMERIC), 0)
      + COALESCE(SAFE_CAST(CHNL_AMT AS NUMERIC), 0)
      + COALESCE(SAFE_CAST(DISTRBTR_AMT AS NUMERIC), 0)) AS gross_revenue
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_DSBD_SCP_RVN_d5f9595b_ec1_cntry_AM`
  WHERE COALESCE(
      SAFE_CAST(CAST(RVN_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(RVN_DD AS STRING), r'[^0-9]', ''))
    ) BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
  GROUP BY
    month_key,
    country_code,
    sale_channel_name,
    product_type,
    service_product_flag,
    device_model_name,
    payment_type_code
)
SELECT
  COALESCE(sales.month_key, revenue.month_key) AS month_key,
  COALESCE(sales.country_code, revenue.country_code) AS country_code,
  COALESCE(sales.sale_channel_name, revenue.sale_channel_name) AS sale_channel_name,
  COALESCE(sales.product_type, revenue.product_type) AS product_type,
  COALESCE(sales.service_product_flag, revenue.service_product_flag) AS service_product_flag,
  COALESCE(sales.device_model_name, revenue.device_model_name) AS device_model_name,
  COALESCE(sales.payment_type_code, revenue.payment_type_code) AS payment_type_code,
  sales.sales_volume,
  revenue.gross_revenue,
  SAFE_DIVIDE(revenue.gross_revenue, NULLIF(sales.sales_volume, 0)) AS average_revenue_per_policy,
  sales.month_key IS NULL AS missing_from_sales,
  revenue.month_key IS NULL AS missing_from_revenue
FROM sales
FULL OUTER JOIN revenue
  ON sales.month_key = revenue.month_key
 AND sales.country_code = revenue.country_code
 AND sales.sale_channel_name = revenue.sale_channel_name
 AND sales.product_type = revenue.product_type
 AND sales.service_product_flag = revenue.service_product_flag
 AND sales.device_model_name = revenue.device_model_name
 AND sales.payment_type_code = revenue.payment_type_code
ORDER BY
  missing_from_sales DESC,
  missing_from_revenue DESC,
  gross_revenue DESC,
  sales_volume DESC
LIMIT 500;

WITH params AS (
  SELECT DATE '2026-01-01' AS start_date, DATE '2026-01-31' AS end_date
),
policy AS (
  SELECT
    COALESCE(
      SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS DATE),
      DATE(SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS TIMESTAMP)),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(INSR_PLCY_CRT_DT AS STRING), r'[^0-9]', ''))
    ) AS policy_date,
    NULLIF(TRIM(CAST(CP_SRL_NUM AS STRING)), '') AS policy_id,
    NULLIF(TRIM(CAST(INSR_PLCY_ID AS STRING)), '') AS insurance_policy_id,
    NULLIF(TRIM(CAST(USR_ID AS STRING)), '') AS user_id,
    NULLIF(TRIM(CAST(PROD_ID AS STRING)), '') AS product_id,
    NULLIF(TRIM(CAST(DVCE_UNQ_ID AS STRING)), '') AS device_unique_id,
    NULLIF(TRIM(CAST(cnty_cd_AM_cntry AS STRING)), '') AS country_code,
    NULLIF(TRIM(CAST(CP_STS_CD AS STRING)), '') AS policy_status_code,
    NULLIF(TRIM(CAST(DVCE_MODEL_NM AS STRING)), '') AS device_model_name,
    NULLIF(TRIM(CAST(SALE_CHNL_CD AS STRING)), '') AS sale_channel_code,
    NULLIF(TRIM(CAST(PMT_TP_CD AS STRING)), '') AS payment_type_code
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_PRTY_SVC_ec1_cntry_AM`
),
policy_dupes AS (
  SELECT
    policy_id,
    COUNT(*) AS source_rows
  FROM policy
  WHERE policy_id IS NOT NULL
  GROUP BY policy_id
  HAVING COUNT(*) > 1
)
SELECT
  'FactPolicySales' AS model_table,
  COUNT(*) AS rows_checked,
  COUNTIF(policy_id IS NULL) AS blank_policy_id_rows,
  COUNTIF(insurance_policy_id IS NULL) AS blank_insurance_policy_id_rows,
  COUNTIF(user_id IS NULL) AS blank_user_id_rows,
  COUNTIF(product_id IS NULL) AS blank_product_id_rows,
  COUNTIF(country_code IS NULL) AS blank_country_code_rows,
  COUNTIF(policy_status_code IS NULL) AS blank_policy_status_rows,
  COUNTIF(device_model_name IS NULL) AS blank_device_model_rows,
  COUNTIF(sale_channel_code IS NULL) AS blank_sale_channel_rows,
  COUNTIF(payment_type_code IS NULL) AS blank_payment_type_rows,
  COUNTIF(policy_date IS NULL) AS invalid_policy_date_rows,
  (SELECT COUNT(*) FROM policy_dupes) AS duplicate_policy_id_groups
FROM policy
JOIN params ON TRUE
WHERE policy_date BETWEEN params.start_date AND params.end_date;

-- ======================================================================
-- 11. History Coverage For Prior Period And Prior Year Measures
-- ======================================================================
-- Prior-year measures are publishable only when both the current period and
-- matching prior-year period exist for the fact used by the measure.

WITH date_ranges AS (
  SELECT
    'FactPolicyDailyAggregate' AS model_table,
    MIN(COALESCE(
      SAFE_CAST(CAST(STAT_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(STAT_DD AS STRING), r'[^0-9]', ''))
    )) AS min_business_date,
    MAX(COALESCE(
      SAFE_CAST(CAST(STAT_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(STAT_DD AS STRING), r'[^0-9]', ''))
    )) AS max_business_date
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_DSBD_SCP_STAT_CNTRCT_d5f9595b_ec1_cntry_AM`

  UNION ALL
  SELECT
    'FactPolicySales',
    MIN(COALESCE(
      SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS DATE),
      DATE(SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS TIMESTAMP)),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(INSR_PLCY_CRT_DT AS STRING), r'[^0-9]', ''))
    )),
    MAX(COALESCE(
      SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS DATE),
      DATE(SAFE_CAST(CAST(INSR_PLCY_CRT_DT AS STRING) AS TIMESTAMP)),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(INSR_PLCY_CRT_DT AS STRING), r'[^0-9]', ''))
    ))
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_PRTY_SVC_ec1_cntry_AM`

  UNION ALL
  SELECT
    'FactRevenue',
    MIN(COALESCE(
      SAFE_CAST(CAST(RVN_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(RVN_DD AS STRING), r'[^0-9]', ''))
    )),
    MAX(COALESCE(
      SAFE_CAST(CAST(RVN_DD AS STRING) AS DATE),
      SAFE.PARSE_DATE('%Y%m%d', REGEXP_REPLACE(CAST(RVN_DD AS STRING), r'[^0-9]', ''))
    ))
  FROM `bigdata-platform-data.scp_7cu`.`SMCDB_TCMC_DSBD_SCP_RVN_d5f9595b_ec1_cntry_AM`
)
SELECT
  model_table,
  min_business_date,
  max_business_date,
  min_business_date <= DATE '2025-01-01'
    AND max_business_date >= DATE '2026-01-31' AS can_support_january_2026_prior_year,
  min_business_date <= DATE '2025-12-01'
    AND max_business_date >= DATE '2026-01-31' AS can_support_january_2026_prior_month
FROM date_ranges
ORDER BY model_table;
