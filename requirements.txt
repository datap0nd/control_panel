	ref table Measures_table

		measure 'Active Policies' =
				VAR _AsOfDate =
					MAXX (
						FILTER ( ALL ( Sales ), NOT ISBLANK ( Sales[Reporting Date] ) ),
						Sales[Reporting Date]
					)
				RETURN
					COUNTROWS (
						FILTER (
							ALL ( Policy ),
							VAR _CoverageStart =
								COALESCE ( Policy[Warranty Start Date], Policy[Policy Created Date] )
							VAR _CoverageEnd = Policy[Warranty End Date]
							VAR _CancellationDate = Policy[Cancellation Date]
							RETURN
								Policy[Policy Status Code Key] = "1"
									&& NOT ISBLANK ( _CoverageStart )
									&& _CoverageStart <= _AsOfDate
									&& ( ISBLANK ( _CoverageEnd ) || _CoverageEnd >= _AsOfDate )
									&& ( ISBLANK ( _CancellationDate ) || _CancellationDate > _AsOfDate )
						)
					)
			formatString: #,##0
			displayFolder: Sales

		measure 'Active Monthly Subscribers' =
				VAR _AsOfDate =
					MAXX (
						FILTER ( ALL ( Sales ), NOT ISBLANK ( Sales[Reporting Date] ) ),
						Sales[Reporting Date]
					)
				RETURN
					COUNTROWS (
						FILTER (
							ALL ( Policy ),
							VAR _CoverageStart =
								COALESCE ( Policy[Warranty Start Date], Policy[Policy Created Date] )
							VAR _CoverageEnd = Policy[Warranty End Date]
							VAR _CancellationDate = Policy[Cancellation Date]
							RETURN
								COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
									&& Policy[Policy Status Code Key] = "1"
									&& NOT ISBLANK ( _CoverageStart )
									&& _CoverageStart <= _AsOfDate
									&& ( ISBLANK ( _CoverageEnd ) || _CoverageEnd >= _AsOfDate )
									&& ( ISBLANK ( _CancellationDate ) || _CancellationDate > _AsOfDate )
						)
					)
			formatString: #,##0
			displayFolder: Sales\Policy Status

		measure 'Monthly Cancellation Events - Selected Months' =
				VAR _AsOfDate =
					MAXX (
						FILTER ( ALL ( Sales ), NOT ISBLANK ( Sales[Reporting Date] ) ),
						Sales[Reporting Date]
					)
				VAR _PeriodLabel = SELECTEDVALUE ( 'Date Filter'[Period Label], "YTD" )
				VAR _SelectedStart =
					MINX (
						CALCULATETABLE ( VALUES ( 'Date Filter'[Date] ), REMOVEFILTERS ( 'Date' ) ),
						'Date Filter'[Date]
					)
				VAR _SelectedEnd =
					MAXX (
						CALCULATETABLE ( VALUES ( 'Date Filter'[Date] ), REMOVEFILTERS ( 'Date' ) ),
						'Date Filter'[Date]
					)
				VAR _PeriodStart =
					SWITCH (
						_PeriodLabel,
						"YTD", DATE ( YEAR ( _AsOfDate ), 1, 1 ),
						"Current Month", DATE ( YEAR ( _AsOfDate ), MONTH ( _AsOfDate ), 1 ),
						"Last 12 Months", EOMONTH ( _AsOfDate, -12 ) + 1,
						"Last Year", DATE ( YEAR ( _AsOfDate ) - 1, 1, 1 ),
						_SelectedStart
					)
				VAR _PeriodEnd =
					SWITCH (
						_PeriodLabel,
						"YTD", _AsOfDate,
						"Current Month", _AsOfDate,
						"Last 12 Months", _AsOfDate,
						"Last Year", DATE ( YEAR ( _AsOfDate ) - 1, 12, 31 ),
						_SelectedEnd
					)
				VAR _FirstMonth = DATE ( YEAR ( _PeriodStart ), MONTH ( _PeriodStart ), 1 )
				VAR _LastMonth = DATE ( YEAR ( _PeriodEnd ), MONTH ( _PeriodEnd ), 1 )
				VAR _MonthCount = DATEDIFF ( _FirstMonth, _LastMonth, MONTH )
				RETURN
					SUMX (
						GENERATESERIES ( 0, _MonthCount, 1 ),
						VAR _MonthStart = EDATE ( _FirstMonth, [Value] )
						VAR _EventStart = MAX ( _PeriodStart, _MonthStart )
						VAR _EventEnd = MIN ( _PeriodEnd, EOMONTH ( _MonthStart, 0 ) )
						RETURN
							COUNTROWS (
								FILTER (
									ALL ( Policy ),
									COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
										&& NOT ISBLANK ( Policy[Cancellation Date] )
										&& Policy[Cancellation Date] >= _EventStart
										&& Policy[Cancellation Date] <= _EventEnd
								)
							)
					)
			formatString: #,##0
			isHidden
			displayFolder: Sales\Policy Status\Diagnostics

		measure 'Monthly Opening Subscription Exposure - Selected Months' =
				VAR _AsOfDate =
					MAXX (
						FILTER ( ALL ( Sales ), NOT ISBLANK ( Sales[Reporting Date] ) ),
						Sales[Reporting Date]
					)
				VAR _PeriodLabel = SELECTEDVALUE ( 'Date Filter'[Period Label], "YTD" )
				VAR _SelectedStart =
					MINX (
						CALCULATETABLE ( VALUES ( 'Date Filter'[Date] ), REMOVEFILTERS ( 'Date' ) ),
						'Date Filter'[Date]
					)
				VAR _SelectedEnd =
					MAXX (
						CALCULATETABLE ( VALUES ( 'Date Filter'[Date] ), REMOVEFILTERS ( 'Date' ) ),
						'Date Filter'[Date]
					)
				VAR _PeriodStart =
					SWITCH (
						_PeriodLabel,
						"YTD", DATE ( YEAR ( _AsOfDate ), 1, 1 ),
						"Current Month", DATE ( YEAR ( _AsOfDate ), MONTH ( _AsOfDate ), 1 ),
						"Last 12 Months", EOMONTH ( _AsOfDate, -12 ) + 1,
						"Last Year", DATE ( YEAR ( _AsOfDate ) - 1, 1, 1 ),
						_SelectedStart
					)
				VAR _PeriodEnd =
					SWITCH (
						_PeriodLabel,
						"YTD", _AsOfDate,
						"Current Month", _AsOfDate,
						"Last 12 Months", _AsOfDate,
						"Last Year", DATE ( YEAR ( _AsOfDate ) - 1, 12, 31 ),
						_SelectedEnd
					)
				VAR _FirstMonth = DATE ( YEAR ( _PeriodStart ), MONTH ( _PeriodStart ), 1 )
				VAR _LastMonth = DATE ( YEAR ( _PeriodEnd ), MONTH ( _PeriodEnd ), 1 )
				VAR _MonthCount = DATEDIFF ( _FirstMonth, _LastMonth, MONTH )
				RETURN
					SUMX (
						GENERATESERIES ( 0, _MonthCount, 1 ),
						VAR _MonthStart = EDATE ( _FirstMonth, [Value] )
						RETURN
							COUNTROWS (
								FILTER (
									ALL ( Policy ),
									VAR _CoverageStart =
										COALESCE ( Policy[Warranty Start Date], Policy[Policy Created Date] )
									VAR _CoverageEnd = Policy[Warranty End Date]
									VAR _CancellationDate = Policy[Cancellation Date]
									RETURN
										COALESCE ( Policy[Product Monthly Amount], 0 ) > 0
											&& NOT ISBLANK ( _CoverageStart )
											&& _CoverageStart <= _MonthStart
											&& ( ISBLANK ( _CoverageEnd ) || _CoverageEnd >= _MonthStart )
											&& ( ISBLANK ( _CancellationDate ) || _CancellationDate > _MonthStart )
								)
							)
					)
			formatString: #,##0
			isHidden
			displayFolder: Sales\Policy Status\Diagnostics

		measure 'Monthly Cancellation Rate' =
				DIVIDE (
					[Monthly Cancellation Events - Selected Months],
					[Monthly Opening Subscription Exposure - Selected Months]
				)
			formatString: 0.0%
			displayFolder: Sales\Policy Status
