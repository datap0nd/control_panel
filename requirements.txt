-- SMP amount feed by subsidiary, segment, and device.
--
-- Source:
--   bi_reporting.inflow_outflow_mv
--
-- Product mapping:
--   bi_reporting.item_mapping_mv
--
-- Output:
--   CSV to stdout. Run with psql and redirect stdout to a file, for example:
--     psql "$DATABASE_URL" -f queries_ran/smp_amount_feed.sql > smp_amount_feed.csv
--
-- Grain:
--   One row per subsidiary, device_type, aggregation_level, segment, and device.
--
-- aggregation_level prevents double counting:
--   - SMP Total: one total smartphone row per subsidiary
--   - Segment: one Flagship and one Mass row per subsidiary
--   - Device: one row per mapped device per subsidiary, including S25F when
--     item_mapping_mv exposes it as the Series value
--
-- Segment rule:
--   - Flagship: item_mapping_mv."Seg1" = 'FLAGSHIP'
--   - Mass: every other SMART item. This keeps SMP Total equal to Flagship
--     plus Mass instead of silently dropping other smartphone Seg1 values.
--
-- Amount definitions come directly from the final MV:
--   inflow_amount  = sum("Total Inflow Amt")
--   outflow_amount = sum("Total Outflow Amt")

copy (
with item_map as (
    select
        "Item ID"::text as item_id,
        max("Seg1"::text) as seg1,
        min(nullif(nullif("Series"::text, ''), 'NAN')) as series
    from bi_reporting.item_mapping_mv
    where "Biz Group" = 'SMART'
    group by
        "Item ID"::text
),
base as (
    select
        f."Subsidiary"::text as subsidiary,
        'Smartphone'::text as device_type,
        case
            when im.seg1 = 'FLAGSHIP' then 'Flagship'
            else 'Mass'
        end as segment,
        coalesce(im.series, nullif(nullif(f."Series"::text, ''), 'NAN'), 'Unmapped') as device,
        coalesce(f."Total Inflow Amt", 0) as inflow_amount,
        coalesce(f."Total Outflow Amt", 0) as outflow_amount
    from bi_reporting.inflow_outflow_mv f
    join item_map im
        on im.item_id = f."ItemID"
    where f."Biz Grp" = 'SMART'
      and f."Subsidiary" is not null
      and f."Subsidiary" <> ''
),
feed_rows as (
    select
        subsidiary,
        device_type,
        'SMP Total'::text as aggregation_level,
        'SMP Total'::text as segment,
        'All Smartphones'::text as device,
        sum(inflow_amount) as inflow_amount,
        sum(outflow_amount) as outflow_amount
    from base
    group by
        subsidiary,
        device_type

    union all

    select
        subsidiary,
        device_type,
        'Segment'::text as aggregation_level,
        segment,
        'All ' || segment as device,
        sum(inflow_amount) as inflow_amount,
        sum(outflow_amount) as outflow_amount
    from base
    group by
        subsidiary,
        device_type,
        segment

    union all

    select
        subsidiary,
        device_type,
        'Device'::text as aggregation_level,
        segment,
        device,
        sum(inflow_amount) as inflow_amount,
        sum(outflow_amount) as outflow_amount
    from base
    group by
        subsidiary,
        device_type,
        segment,
        device
)
select
    subsidiary,
    device_type,
    aggregation_level,
    segment,
    device,
    inflow_amount,
    outflow_amount
from feed_rows
order by
    subsidiary,
    case aggregation_level
        when 'SMP Total' then 1
        when 'Segment' then 2
        when 'Device' then 3
        else 9
    end,
    case segment
        when 'SMP Total' then 1
        when 'Flagship' then 2
        when 'Mass' then 3
        else 9
    end,
    device
) to stdout with csv header;
